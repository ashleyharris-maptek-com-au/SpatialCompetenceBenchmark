// PHAROH'S PYRAMID CONSTRUCTION PLAN
// ------------------------------------
// Units: Meters (SI)
//
// CONSTRUCTION PARAMETERS:
// Total Stone Budget: 2,500,000 blocks
// Block Dimensions: 0.5m x 0.5m x 0.5m
// Block Volume: 0.125 m^3
// Labor Constraint: 10 men/stone (Feasibility check only, material is limiting factor)
//
// STRUCTURAL CALCULATIONS:
// Pyramid Logic: To ensure 4 corners of a block sit on the centers of blocks below,
// the pyramid must follow a sequence of squares where layer L has side length L blocks.
// Sum of squares: Sum(k^2) for k=1 to H = H(H+1)(2H+1)/6
//
// Solving for H where Sum <= 2,500,000:
// Max Height (Layers) H = 195
// Stones Used: 2,490,410
// Stones Remaining: 9,590
//
// DIMENSIONS:
// Pyramid Height (Core): 195 layers * 0.5m = 97.5m
// Core Base Width: 195 blocks * 0.5m = 97.5m
//
// FACADE GEOMETRY:
// The marble facade must cover the steps to create a smooth surface.
// To cover the top-outer edge of every step (from the 97.5m base to the 0.5m top),
// the smooth pyramid must extend slightly beyond the core.
//
// Fitting Calculation:
// Top of Layer 195 (Top Block): z = 97.5m, Width = 0.5m
// Top of Layer 1 (Base Block): z = 0.5m, Width = 97.5m
//
// A linear slope passing through (Width=0.5, z=97.5) and (Width=97.5, z=0.5):
// Rise = 97.0m, Run (on diameter) = 97.0m. Slope = 1:1 (Height : Width).
// Note: Slope relative to center (radius) is 2:1.
//
// Geometric Peak (z_max): 98.0m
// Facade Base Width (at z=0): 98.0m
//
// VOLUMES:
// Core Volume: 311,301.250 m^3
// Marble Facade Total Volume (Solid Pyramid): 313,726.667 m^3
// Marble Required (Facade - Core): ~2,425.417 m^3

$fn = 4; // Set resolution to 4 for square pyramid generation

// --- CONFIGURATION ---
block_size = 0.5; // meters
num_layers = 195;

// Calculated dimensions for the smooth outer shell
facade_base_width = (num_layers + 1) * block_size; // 98.0m
facade_height = (num_layers + 1) * block_size;     // 98.0m

// --- RENDERING ---

// 1. The Stone Core (Stepped Pyramid)
// Rendered in Sandstone color
module stone_core() {
    color([0.9, 0.75, 0.5]) 
    union() {
        for (i = [1 : num_layers]) {
            // i=1 is the top tip (1x1 block)
            // i=195 is the bottom base (195x195 blocks)
            
            layer_width = i * block_size;
            
            // Elevation: Top layer is highest.
            // i=1 is at z = (195-1)*0.5 = 97.0
            z_pos = (num_layers - i) * block_size;
            
            translate([0, 0, z_pos])
                // Center the layer so coordinate 0,0 is the tomb center
                translate([-layer_width/2, -layer_width/2, 0])
                cube([layer_width, layer_width, block_size]);
        }
    }
}

// 2. The Marble Facade (Smooth Casing)
// Rendered in White Marble (semi-transparent to show core)
// This represents the final smooth surface required by the stonemason.
module marble_facade() {
    color([1, 1, 1, 0.3]) 
    translate([0, 0, 0])
    rotate([0, 0, 45]) // Rotate 45 deg because cylinder($fn=4) creates a diamond shape
    cylinder(
        h = facade_height, 
        r1 = (facade_base_width / 2) * sqrt(2), // Radius to corner of square base
        r2 = 0, 
        $fn = 4
    );
}

// --- ASSEMBLY ---

// Group for easy viewing
union() {
    stone_core();
    marble_facade();
}
