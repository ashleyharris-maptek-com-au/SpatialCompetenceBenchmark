// Tallest stable stepped pyramid from 150,000 cubes of side 50 mm.
// Constraint: each cube on an upper layer must have its 4 bottom corners
// above the interior of cubes below => requires a 1/2-cube inset per side per layer.
// This yields layer n as an N-by-N grid of cubes, with N decreasing by 1 each layer.
// Max height occurs when total cubes used <= 150,000.
//
// Let base layer have N cubes per side. Total cubes = sum_{k=1..N} k^2 = N(N+1)(2N+1)/6.
// Largest N with sum <= 150000 is N=76 (sum=148,204). Next N=77 is 152,075 (>150,000).
// Height = 76 layers, each 50 mm => 3,800 mm.

$fn = 24; // not used for cubes, but kept modest for CAD friendliness

// --- Parameters (all in mm) ---
cube_side = 50;          // each stone is a 50mm cube
N_base    = 76;          // cubes per side at base layer
layers    = N_base;      // number of layers
offset    = cube_side/2; // 1/2 cube inset per side per layer for corner-support rule

// --- Derived ---
base_size = N_base * cube_side; // overall base footprint (mm)
height    = layers * cube_side; // overall height (mm)

// North is +Y; pyramid centered over tomb at origin (0,0).
// We build as a union of layer slabs (each slab is a solid rectangular prism),
// equivalent to placing all cubes without internal voids, but vastly faster to render.

// Layer i (0-based from bottom):
// side length = (N_base - i) * cube_side
// z range: [i*cube_side, (i+1)*cube_side)
// x,y centered, with inset = i*offset from base alignment.
module stepped_pyramid(N=N_base, a=cube_side) {
    for (i = [0 : N-1]) {
        side = (N - i) * a;
        z0   = i * a;
        translate([-side/2, -side/2, z0])
            cube([side, side, a], center=false);
    }
}

stepped_pyramid();
