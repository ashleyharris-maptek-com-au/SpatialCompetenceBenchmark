// Tallest pyramid buildable with available stones
// Units: meters (SI). Dimensions given to millimetre precision where appropriate.
// Each large stone is a cube with side length stone_side = 0.500 m (50 cm).
// Stacking rule => each upper layer has one fewer stone per side than the layer below,
// so the pyramid layers are square grids of sizes n x n, (n-1) x (n-1), ..., 1 x 1.
//
// Available stones: 2,500,000
// Stones required for an n-wide base: sum_{k=1..n} k^2 = n*(n+1)*(2*n+1)/6
//
// The largest integer n with required_stones <= 2,500,000 is n = 195
// required_stones = 2,490,670 stones
// unused_stones = 2,500,000 - 2,490,670 = 9,330 stones
//
// Stone volume = (0.5 m)^3 = 0.125 m^3
// Total solid volume used = 2,490,670 * 0.125 = 311,333.750 m^3
//
// Pyramid base side length = n * stone_side = 195 * 0.500 = 97.500 m
// Pyramid total height = n * stone_side = 97.500 m
//
// Orientation: pyramid faces are aligned with the X/Y axes.
// +Y is north; therefore the face at +Y faces north. The pyramid is centred at the origin (0,0)
// so it is centred over the future tomb at the origin.
//
// The model below constructs the stepped pyramid as a stack of solid horizontal slabs,
// one slab per stone-layer. Each slab exactly matches the continuous footprint of that layer
// (k stones per side -> slab of size k*stone_side by k*stone_side, thickness stone_side).
// This yields a precise external stepped surface (each step corresponds to a stone layer).
// The mesh is optimized (195 solid cuboids) for CAD review while remaining precise to 1 mm.

stone_side = 0.500;       // m (50 cm)
n_layers = 195;           // computed maximum base width in stones
// derived (for reference)
stones_used = n_layers*(n_layers+1)*(2*n_layers+1)/6;
total_volume_m3 = stones_used * (stone_side^3);

// Visual helper: adjust if you want to view only a portion for speed
show_full_pyramid = true;

// Draw the stepped pyramid
union() {
    if (show_full_pyramid) {
        // build from bottom (k = n_layers) up to top (k = 1)
        for (k = [n_layers : -1 : 1]) {
            layer_size = k * stone_side;                    // meters
            z = (n_layers - k) * stone_side;               // bottom of this slab
            translate([-layer_size/2, -layer_size/2, z])
                cube([layer_size, layer_size, stone_side], center = false);
        }
    }
}

// Optional axes and ground plane for orientation and masonry reference
// (comment out if not required in final CAD)
module axes(length=10, thickness=0.02) {
    // X axis (red), Y axis (green), Z axis (blue)
    color([1,0,0]) translate([0,0,0]) cube([length, thickness, thickness], center=false);
    color([0,1,0]) translate([0,0,0]) cube([thickness, length, thickness], center=false);
    color([0,0,1]) translate([0,0,0]) cube([thickness, thickness, length], center=false);
}
