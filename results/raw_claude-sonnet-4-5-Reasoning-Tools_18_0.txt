```openscad
stone_size = 500;
n_layers = 38;
base_width = n_layers * stone_size;
total_height = n_layers * stone_size;

module stepped_pyramid() {
    translate([-base_width/2, -base_width/2, 0]) {
        for (layer = 0; layer < n_layers; layer++) {
            stones_per_side = n_layers - layer;
            layer_width = stones_per_side * stone_size;
            offset = layer * stone_size / 2;
            translate([offset, offset, layer * stone_size])
                cube([layer_width, layer_width, stone_size]);
        }
    }
}

module marble_cladding() {
    half_base = base_width / 2;
    polyhedron(
        points = [
            [-half_base, -half_base, 0],
            [half_base, -half_base, 0],
            [half_base, half_base, 0],
            [-half_base, half_base, 0],
            [0, 0, total_height]
        ],
        faces = [
            [0, 1, 2, 3],
            [0, 4, 1],
            [1, 4, 2],
            [2, 4, 3],
            [3, 4, 0]
        ]
    );
}

stepped_pyramid();
%marble_cladding();
```
