// Tallest stable stepped pyramid using 2,500,000 stones of 0.5 m cubes
// Units: millimeters (SI)
// Constraint: each upper cube must sit with its 4 bottom corners at the centers of 4 cubes below
// => Each higher layer is offset by 0.25 m in X and Y and shrinks by 1 cube per side per layer.
// Total stones for L layers with base N x N:
// S(L,N) = sum_{k=0..L-1} (N-k)^2 = L*N^2 - N*L*(L-1) + (L-1)*L*(2L-1)/6
// Max L with S <= 2,500,000 is achieved with L=145 and N=145 (exactly N=L), giving S=1,024,360.

$fn = 24;

cube_side = 500;        // mm
half_step = cube_side/2; // 250 mm offset per level (corner-to-center rule)

layers = 145;
base_n  = 145;

// Build as a stack of solid slabs (one per layer) for CAD efficiency,
// while remaining exact to the millimeter and preserving all step edges.
module stepped_pyramid(L, N, s, dx) {
    for (k = [0 : L-1]) {
        n = N - k;
        translate([k*dx, k*dx, k*s])
            cube([n*s, n*s, s], center=false);
    }
}

// Centered over the tomb at origin; north is +Y (orientation is symmetric).
translate([-base_n*cube_side/2, -base_n*cube_side/2, 0])
    stepped_pyramid(layers, base_n, cube_side, half_step);
