// Pyramid specifications
// Stone: 50cm (0.5m) cubes
// Constraint: Each stone's 4 base corners must sit in middle of stones below
// This means stones are offset by 0.25m (half stone width) in each layer
// Available stones: 150,000
// Available men: 100,000 (each stone needs 10 men, so max 10,000 stones can be moved at once)

// Layer stone count calculation:
// Layer 0 (base): n×n stones
// Layer 1: (n-1)×(n-1) stones (offset by 0.25m on each side)
// Layer k: (n-k)×(n-k) stones
// Total stones for pyramid of base n: sum from k=0 to n-1 of (n-k)^2 = n(n+1)(2n+1)/6

// Finding maximum n where n(n+1)(2n+1)/6 <= 150,000
// n=62: 62*63*125/6 = 81,375
// n=63: 63*64*127/6 = 85,344
// n=64: 64*65*129/6 = 89,440
// n=65: 65*66*131/6 = 93,665
// n=66: 66*67*133/6 = 98,021
// n=67: 67*68*135/6 = 102,510
// n=68: 68*69*137/6 = 107,134
// n=69: 69*70*139/6 = 111,895
// n=70: 70*71*141/6 = 116,795
// n=71: 71*72*143/6 = 121,836
// n=72: 72*73*145/6 = 127,020
// n=73: 73*74*147/6 = 132,349
// n=74: 74*75*149/6 = 137,825
// n=75: 75*76*151/6 = 143,450
// n=76: 76*77*153/6 = 149,226

// Maximum base size: 76×76 stones

module pyramid_smooth() {
    stone_size = 500; // 50cm in mm
    base_n = 76;
    offset = stone_size / 2; // 250mm offset per layer
    
    // Calculate pyramid points
    base_width = base_n * stone_size;
    height = base_n * offset;
    
    // Smooth pyramid hull
    polyhedron(
        points = [
            // Base corners
            [0, 0, 0],
            [base_width, 0, 0],
            [base_width, base_width, 0],
            [0, base_width, 0],
            // Apex
            [base_width/2, base_width/2, height]
        ],
        faces = [
            [0, 1, 2, 3], // base
            [0, 4, 1],    // south face (faces +y is north)
            [1, 4, 2],    // east face
            [2, 4, 3],    // north face
            [3, 4, 0]     // west face
        ]
    );
}

// Center pyramid over tomb at origin, facing north (+y)
translate([-19000, -19000, 0])
    pyramid_smooth();
