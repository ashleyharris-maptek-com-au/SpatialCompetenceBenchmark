// Pharaoh's pyramid planning: tallest possible with 2,500,000 stones of 50 cm cubes.
// All units are millimeters (mm). Final shape is a stepped square pyramid.
// Each layer k from bottom has (n - k) x (n - k) stones; layer thickness = 50 cm = 500 mm.

STONE_SIDE_MM = 500;      // 50 cm cube stones
TOTAL_STONES   = 2500000;  // budget

// Compute maximum number of layers n such that sum_{i=1..n} i^2 <= TOTAL_STONES
n = 0;
sum_squares = 0;

// Use a safe upper bound; for 2.5e6 stones, n is around 195.
for (i = [1 : 1000]) {
    if (sum_squares + i*i <= TOTAL_STONES) {
        sum_squares = sum_squares + i*i;
        n = i;
    } else {
        break;
    }
}

// If something went wrong, fall back to a known safe value
if (n == 0) n = 195;  // fallback, matches manual calculation

// Optional: visualize with a distinct color by layer (comment out if not desired)
//n = 195; // force fallback visualization (uncomment if needed)

// Build the pyramid as n stacked layers from bottom (n x n) to top (1 x 1)
$fn = 64; // smoothness for any future spherical elements (not required for cubes)

// Loop over layers: bottom layer (layer=0) has n x n stones; top layer (layer=n-1) has 1x1
for (layer = [0 : n-1]) {
    m = n - layer;                 // number of stones along one side for this layer
    L = m * STONE_SIDE_MM;         // footprint side length of this layer (mm)
    z = layer * STONE_SIDE_MM;     // vertical position of this layer's bottom (mm)

    translate([-L/2, -L/2, z])
        cube([L, L, STONE_SIDE_MM], center = false);
}
