// Pyramid Plan for Pharaoh's Tomb
// Units: meters (m)
// Each stone: 0.5m cube
// Conventions:
//   +y is North
//   Pyramid centered over tomb at (0,0) on ground plane z=0

// -------------------- PARAMETERS --------------------

stone = 0.5;          // edge length of each stone cube (m)
men_per_stone = 10;   // men required per stone (for reference)
total_men = 100000;   // total men available (for reference)
total_stones = 600000;

// Determine maximum whole-layer height H such that H^3 <= total_stones
// H^3 <= 600000
// H = floor(cuberoot(600000)) = 84 because 84^3 = 592,704 and 85^3 = 614,125
H = 84;               // number of stone-layers in height
stones_used = H*H*H;  // 592,704 (for reference, > 600,000 would be impossible)

// Derived continuous geometry parameters
base_half = H*stone/2;      // half-width of base (m)
pyr_height = H*stone;       // total pyramid height (m)

// -------------------- GEOMETRY OVERVIEW --------------------
// The discrete stone arrangement that obeys the “corners on centers” rule
// and has no internal voids fully occupies a regular step pyramid that
// is well-approximated by a solid right square pyramid of:
//   base side: H*stone
//   height:    H*stone
//
// This OpenSCAD model provides:
//   1. A solid right square pyramid for accurate external marble cutting.
//   2. An optional stepped-stone visualization that matches the cubic count,
//      kept lightweight enough for CAD viewing.

// -------------------- CORE SHAPES --------------------

// Continuous smooth pyramid for marble planning
module marble_pyramid() {
    // A right square pyramid with base cube aligned to axes
    // Base: square in x-y, centered at origin, on z=0
    // Apex at (0,0,pyr_height)
    polyhedron(
        points = [
            [ base_half,  base_half, 0],   // 0
            [-base_half,  base_half, 0],   // 1
            [-base_half, -base_half, 0],   // 2
            [ base_half, -base_half, 0],   // 3
            [ 0,          0,         pyr_height]  // 4 apex
        ],
        faces = [
            [0,1,2,3], // base (downward)
            [0,3,4],   // south face (+x,-y)
            [3,2,4],   // west  face (-x,-y)
            [2,1,4],   // north face (-x,+y)
            [1,0,4]    // east  face (+x,+y)
        ]
    );
}

// Discrete stepped pyramid for visual reference / checking
// This matches the layerwise H x H x H cube filling.
module stone_step_pyramid(max_layers = H) {
    // To keep this viewable, we allow the stonemason to
    // show fewer layers if desired via max_layers.
    layer_limit = min(max_layers, H);

    for (z_layer = [0:layer_limit-1]) {
        n = H - z_layer;  // stones per side on this layer
        z = (z_layer + 0.5)*stone;

        // layer footprint half-width for these n stones
        half = n*stone/2;

        // approximate each layer as a single slab (union of cubes),
        // to stay lightweight for CAD.
        translate([0,0,z])
            cube([2*half, 2*half, stone], center=true);
    }
}

// -------------------- OUTPUT --------------------

// For marble cutting: use only the smooth pyramid.
// Comment/uncomment according to need.

marble_pyramid();
