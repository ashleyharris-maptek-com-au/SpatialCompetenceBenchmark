// Pyramid specifications
// Stone: 50cm (0.5m) cube
// Total stones: 2,500,000
// Constraint: Each stone's 4 base corners must sit in middle of stones below
// This means offset = 0.25m (half stone) between layers

stone_size = 500; // mm (50cm)
half_stone = stone_size / 2;

// Calculate maximum pyramid dimensions
// Layer n (from top, n=1 is apex) has side length: n stones
// Total stones for height h: sum from 1 to h of n^2 = h(h+1)(2h+1)/6
// Solving h(h+1)(2h+1)/6 = 2,500,000
// h ≈ 216 layers

height_layers = 216;
total_height = height_layers * stone_size;
base_size = height_layers * stone_size;

// Create smooth pyramid hull
module smooth_pyramid() {
    polyhedron(
        points = [
            // Base corners
            [-base_size/2, -base_size/2, 0],
            [base_size/2, -base_size/2, 0],
            [base_size/2, base_size/2, 0],
            [-base_size/2, base_size/2, 0],
            // Apex
            [0, 0, total_height]
        ],
        faces = [
            [0, 1, 2, 3],  // base
            [0, 4, 1],      // face
            [1, 4, 2],      // face
            [2, 4, 3],      // face
            [3, 4, 0]       // face
        ]
    );
}

smooth_pyramid();
