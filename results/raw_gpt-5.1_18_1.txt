// Pyramid Plan for Pharaoh - OpenSCAD
// Units: millimetres (mm)

//------------------------------------------------------------------
// PARAMETERS
//------------------------------------------------------------------

// Stone (block) size: 50 cm = 500 mm
block_size = 500;           // mm

// Total available stones
total_stones = 150000;

// Number of men (not limiting here, only stones)
men = 100000;

//------------------------------------------------------------------
// PYRAMID LAYOUT COMPUTATION
//------------------------------------------------------------------
//
// Each "step level" k (k = 1..K) is a complete layer of cubes:
// - Level k has side length (in blocks) = n_k = 2*k - 1
// - Stones in level k: n_k^2 = (2*k - 1)^2
// Total stones for K levels: sum_{k=1..K} (2k-1)^2 = (4/3)K^3 - (1/3)K
// Find largest integer K such that this sum ≤ total_stones.
//
// Solve (4/3)K^3 - (1/3)K ≤ 150000
// For integer K we evaluate exactly by the discrete sum.
//
// Check K = 39:
//   sum = Σ(2k-1)^2, k=1..39 = 79^3 / 12 = 493039
//   493039 > 150000  → too many stones.
//
// Therefore the pyramid can only have K = 38 full levels.
//
// For K = 38: total_stones_used = Σ(2k-1)^2, k=1..38 = 75*76*77/6 = 366100
// This still exceeds 150000, so we scale the pattern down to the largest K
// satisfying the exact discrete sum:
//
// We need the largest integer K s.t. 4K^3 - K ≤ 450000 (multiply both sides by 3)
//
// Check K = 48:
//   4*48^3 - 48 = 4*110592 - 48 = 442368 - 48 = 442320  ( ≤ 450000 ) OK
// Check K = 49:
//   4*49^3 - 49 = 4*117649 - 49 = 470596 - 49 = 470547  ( > 450000 ) too many
//
// Therefore the tallest pyramid with full stepped levels is K = 48.
//
// Total stones used:
//   S = (4K^3 - K)/3 = (4*48^3 - 48)/3 = 442320 / 3 = 147440
// Stones remaining unused: 150000 - 147440 = 2560
//
//------------------------------------------------------------------
// GEOMETRY
//------------------------------------------------------------------
//
// - Levels k = 1..K (K=48).
// - Level k has an (2k-1) × (2k-1) square of blocks.
// - Block size is 500 mm cube.
// - Pyramid is centered in X and Y over (0,0).
// - North is +Y; we align faces accordingly.
//
// Level k is placed at height z = (k-1)*block_size.
// Side (in mm) of level k: L_k = (2k - 1)*block_size.
// Offset in X and Y for centering: from -L_k/2 to +L_k/2.
//
//------------------------------------------------------------------
// MODEL
//------------------------------------------------------------------

// Number of step levels
K = 48;

module pyramid_step_blocks() {
    for (k = [1:K]) {
        n = 2*k - 1;                // blocks per side on this level
        half = n/2;                 // half-count for centering indices

        z = (k - 1) * block_size;   // height of this level (bottom of blocks)

        // Block grid: i,j ∈ [-half : +half], but only integer positions that
        // give exactly n blocks. Use integer indices from 0..(n-1) and shift.
        for (ix = [0:n-1]) {
            for (iy = [0:n-1]) {
                // Centered coordinates (block centers)
                x = (ix - (n-1)/2) * block_size;
                y = (iy - (n-1)/2) * block_size;

                translate([x, y, z])
                    cube(block_size, center = true);
            }
        }
    }
}

//------------------------------------------------------------------
// OUTPUT
//------------------------------------------------------------------

pyramid_step_blocks();
