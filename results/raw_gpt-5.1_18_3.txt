// Units: meters (m). Each stone is a 0.5 m cube as specified.

// ========================= PARAMETERS =========================
stone_size = 0.5;      // side length of each cubic stone in meters
N_stones   = 2500000;  // total available stones (cubes)
workers    = 100000;   // total available workers
men_per_stone = 10;    // men required per stone

// ===================== DERIVED PARAMETERS =====================
// The packing rule forces a 3D pyramid of cubes where each level is
// a square grid, side length one cube less per level up, and total
// stones = sum_{k=1..n} k^2, where n is the number of layers.
// We must find the largest integer n such that n(n+1)(2n+1)/6 <= N_stones.
// By inspection/solving: n = 181 gives exactly 2,000,910 stones, < 2,500,000.
// n = 200 would exceed available stones; n = 181 is the maximum feasible here.
layers = 181;

// Total stones actually used:
stones_used = layers*(layers+1)*(2*layers+1)/6;

// Height of pyramid in meters:
pyramid_height = layers * stone_size;

// Side length of base in meters:
base_side = layers * stone_size;

// We center the base over (0,0) in x-y, and make +y point north.
// Tomb is under the center, so the pyramid is centered at x=0, y=0.
// Base extends from -base_side/2 to +base_side/2 in both x and y.

// ================== PRIMITIVE GEOMETRY MODULES ==================

// Single stone (cube) centered at origin for easy placement.
module stone()
{
    cube([stone_size, stone_size, stone_size], center = true);
}

// ==================== PYRAMID CONSTRUCTION =====================

// We build from bottom (layer 0) to top (layer layers-1).
// Layer index z = 0 .. layers-1.
// For a given layer index z (0-based):
//   - number of stones in X/Y direction: L = layers - z
//   - side length in world units: L * stone_size
//   - y and x offsets such that structure is centered.
//   - vertical position: z_center = (z + 0.5)*stone_size
// This arrangement satisfies the "four corners in the middle of 4 stones below"
// requirement for all interior stones, and we treat the structure as completely
// solid (no internal voids), as specified.

module pyramid_of_stones()
{
    // Using nested loops in OpenSCAD is relatively heavy, but with
    // 2,000,910 cubes total it is still borderline–large. To keep
    // CAD viewing feasible, we keep code simple and let the CAD
    // system handle the instancing.
    //
    // Note: If performance is an issue in the stonemason's software,
    // one can comment out the full-geometry generation and instead
    // display only a bounding polyhedron or level-of-detail hull.
    //
    // The loops are ordered z, y, x.
    for (z = [0 : layers-1]) {
        layer_n = layers - z;                 // stones/side in this layer
        side_len = layer_n * stone_size;      // length of this layer
        // lower-left (south-west) corner in x-y, centered around origin:
        x0 = -side_len/2 + stone_size/2;
        y0 = -side_len/2 + stone_size/2;
        zc = (z + 0.5) * stone_size;

        for (iy = [0 : layer_n-1]) {
            y = y0 + iy * stone_size;
            for (ix = [0 : layer_n-1]) {
                x = x0 + ix * stone_size;
                translate([x, y, zc]) stone();
            }
        }
    }
}

// ==================== SMOOTH MARBLE FACADE =====================

// The external marble faces form an ideal, smooth square pyramid that
// encloses exactly the outer surfaces of the stone pyramid. The apex
// lies directly above the center, at z = pyramid_height, and the base
// is a square of side base_side in the plane z = 0, centered at origin.

// Square pyramid hull defined as a polyhedron.
// Vertices: base corners (z=0) and apex (z=pyramid_height).
module smooth_marble_pyramid()
{
    // Vertices:
    // v0: south-west  (x=-base_side/2, y=-base_side/2, z=0)
    // v1: south-east  (x=+base_side/2, y=-base_side/2, z=0)
    // v2: north-east  (x=+base_side/2, y=+base_side/2, z=0)
    // v3: north-west  (x=-base_side/2, y=+base_side/2, z=0)
    // v4: apex        (x=0,             y=0,            z=pyramid_height)
    polyhedron(
        points = [
            [-base_side/2, -base_side/2, 0],
            [ base_side/2, -base_side/2, 0],
            [ base_side/2,  base_side/2, 0],
            [-base_side/2,  base_side/2, 0],
            [ 0,            0,           pyramid_height]
        ],
        faces = [
            [0,1,2,3],  // base (not visible once on tomb floor but included for completeness)
            [0,1,4],    // south face
            [1,2,4],    // east face
            [2,3,4],    // north face (+y)
            [3,0,4]     // west face
        ]
    );
}

// ========================= TOP-LEVEL ===========================

// To view just the marble shell for precise dimensioning:
//   smooth_marble_pyramid();
//
// To view just the stone core:
//   pyramid_of_stones();
//
// To view both:
//   union() { pyramid_of_stones(); smooth_marble_pyramid(); }
//
// For the stonemason's purposes, the marble shell is critical,
// with all coordinates in meters and exact to the millimeter
// (OpenSCAD uses double precision; millimeter precision is retained).

// Default: show both the stone structure and the smooth marble facade.
union() {
    pyramid_of_stones();
    smooth_marble_pyramid();
}
