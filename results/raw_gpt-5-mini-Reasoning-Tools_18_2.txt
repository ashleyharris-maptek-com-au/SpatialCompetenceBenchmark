// OpenSCAD file (units: meters)
// Tallest pyramid buildable with 600,000 stones (stone = cube 0.5 m side)
// Stones required: sum_{k=1..n} k^2 <= 600000  -> n = 121
// This model presents:
//  - the stepped (stone) pyramid (one cuboid per layer; exact to mm)
//  - the ideal smooth pyramid (continuous outer faces through base edges and apex)
//  - the marble wedge volume = (smooth pyramid) - (stepped pyramid)  (available as an option)
// The stepped pyramid is centered at the origin (tomb) and oriented so +Y is North.

//
// Parameters (SI - meters)
s = 0.5;                 // stone side length (m)
n_base = 121;            // stones per side on base (max given 600000 stones)
stones_budget = 600000;  // available stones

// derived exact integers / measures
stones_used = n_base*(n_base + 1)*(2*n_base + 1)/6;  // sum_{k=1..n_base} k^2
stones_left = stones_budget - stones_used;

W = n_base * s;          // base width (m)
H = n_base * s;          // pyramid height (m); here H == W (meters)
stone_vol = s*s*s;       // volume per stone (m^3)
stepped_volume = stones_used * stone_vol;   // total stepped stone volume (m^3)
smooth_volume = W*W*H/3;                    // continuous pyramid volume (m^3)
marble_volume = stepped_volume - smooth_volume; // total marble needed to cover (m^3)

// Display / export options
show_stepped = true;   // show the stepped stone pyramid (exact steps)
show_smooth  = false;  // show the ideal smooth pyramid (the final facade shape)
show_wedges  = true;   // show the marble wedge volume = smooth minus stepped (single solid)
                      // NOTE: showing wedges (difference operation) is exact and gives the
                      //       marble shell as one solid; stonemason can partition per-stone externally.

// Echo key numbers (printed to OpenSCAD console)
echo("STONE SIDE (m) = ", s);
echo("BASE STONES PER SIDE = ", n_base);
echo("STONES USED = ", stones_used);
echo("STONES LEFT = ", stones_left);
echo("TOTAL STEPPED VOLUME (m^3) = ", stepped_volume);
echo("SMOOTH PYRAMID VOLUME (m^3) = ", smooth_volume);
echo("TOTAL MARBLE VOLUME (m^3) = ", marble_volume);
echo("PYRAMID BASE WIDTH (m) = ", W);
echo("PYRAMID HEIGHT (m) = ", H);

// -----------------------
// Modules
// -----------------------

module stepped_pyramid() {
    // Build each layer as a single cuboid (exactly fills all stones in that layer).
    // Layer index l = 0 (base) .. n_base-1 (top)
    union() {
        for (l = [0 : n_base - 1]) {
            m = n_base - l;                       // stones per side in this layer
            layer_size = m * s;                   // meters
            z = l * s;                            // layer base z (meters)
            translate([-layer_size/2, -layer_size/2, z])
                cube([layer_size, layer_size, s], center = false);
        }
    }
}

module smooth_pyramid() {
    // Continuous pyramid with square base W x W at z=0 and apex at (0,0,H).
    pts = [
        [-W/2, -W/2, 0],
        [ W/2, -W/2, 0],
        [ W/2,  W/2, 0],
        [-W/2,  W/2, 0],
        [  0  ,   0  , H]
    ];
    // faces: apex->base triangles, and base quad (ordered for consistent normals)
    faces = [
        [4,0,1],
        [4,1,2],
        [4,2,3],
        [4,3,0],
        [0,3,2,1]
    ];
    polyhedron(points = pts, faces = faces);
}

// -----------------------
// Assembly / View
// -----------------------

translate([0,0,0])  // centered over tomb at origin
union() {
    if (show_stepped) {
        // stone color (preview only)
        color([0.65,0.65,0.65,1])
            stepped_pyramid();
    }

    if (show_smooth) {
        // smooth face overlay (semi-transparent preview)
        color([1,0.85,0.7,0.45])
            smooth_pyramid();
    }

    if (show_wedges) {
        // Marble wedges = (smooth outer pyramid) minus (stepped stones).
        // Provided as one exact solid (CSG difference). Export this for the stonemason
        // if they want the complete marble shell; they can partition per-stone as needed.
        color([1,0.98,0.95,0.85])
            difference() {
                smooth_pyramid();
                // subtract the stepped pyramid (all stone volumes). This yields the
                // marble shell exactly fitting stones; geometrically precise to mm in coordinates.
                stepped_pyramid();
            }
    }
}

// End of file. Units: meters. All dimensions are integer multiples of 0.5 m (500 mm),
// so coordinates and volumes are precise to the millimeter (0.001 m) within floating-point limits.
