// Tallest stepped stone pyramid core given:
// - 150000 stones, each 0.5 m cube
// - stacking rule: each stone above rests with its four base corners over centers of 4 stones below
// This yields 76 full layers (from 76x76 at base up to 1x1 at top),
// using 149226 stones in total.

stone_size        = 0.5;      // meters, edge length of each stone
levels            = 76;       // number of vertical layers of stones
stones_used       = 149226;   // total stones in this configuration
stones_available  = 150000;
stones_remaining  = stones_available - stones_used;

height_m      = levels * stone_size; // 38.0 m
base_width_m  = levels * stone_size; // 38.0 m

echo("Levels:", levels);
echo("Stones used:", stones_used);
echo("Stones remaining:", stones_remaining);
echo("Pyramid height (m):", height_m);
echo("Pyramid base width (m):", base_width_m);

// Stepped stone core, centered over the tomb at (0,0).
// North is +y; pyramid faces are aligned with the coordinate axes.
module stone_core() {
    for (k = [0 : levels - 1]) {
        translate([
            -(levels - k) * stone_size / 2,
            -(levels - k) * stone_size / 2,
             k * stone_size
        ])
        cube([
            (levels - k) * stone_size,
            (levels - k) * stone_size,
             stone_size
        ], center = false);
    }
}

// Render the core pyramid
stone_core();
