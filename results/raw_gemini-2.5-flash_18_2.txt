// OpenSCAD plan for the Pharaoh's pyramid

// --- Global Parameters ---
stone_size = 0.5; // Each stone is a 50cm (0.5m) side-length cube. (Metric SI units)
num_total_stones_budget = 600000; // Total number of available stones.

// --- Calculation for the Tallest Pyramid ---
// The pyramid's structure follows a specific rule: "each stone must have all 4 base
// corners sitting in the middle of another stone." This implies a step pyramid
// where each layer steps back by half a stone length on each side compared to the layer below.
//
// If the base layer has 'N' stones per side, the layers above it will have (N-1), (N-2), ..., 1 stone per side.
// The total number of stones in such a pyramid is the sum of squares from 1 to N:
// Sum(j=1 to N) j^2 = N * (N + 1) * (2N + 1) / 6
//
// We need to find the largest 'N' (number of stones per side in the base layer)
// such that the total number of stones required does not exceed the budget.
//
// Let's test values for N:
// If N = 121: Total stones = 121 * (121 + 1) * (2 * 121 + 1) / 6
//                           = 121 * 122 * 243 / 6
//                           = 598,641 stones. (This is within the budget)
//
// If N = 122: Total stones = 122 * (122 + 1) * (2 * 122 + 1) / 6
//                           = 122 * 123 * 245 / 6
//                           = 613,845 stones. (This exceeds the budget)
//
// Therefore, the maximum number of stones per side for the base layer is 121.
num_base_stones = 121; // The largest N that fits the budget.
stones_used = 598641; // The actual number of stones used for this pyramid.
stones_remaining = num_total_stones_budget - stones_used; // 1359 stones remaining.

// --- Derived Dimensions ---
// Total height of the pyramid (each layer adds one stone's height)
total_pyramid_height = num_base_stones * stone_size; // 121 * 0.5m = 60.5m

// Base width/depth of the pyramid (N stones wide)
base_width = num_base_stones * stone_size; // 121 * 0.5m = 60.5m

// --- OpenSCAD Model Generation ---
// The pyramid is centered over the origin [0,0,0] (the future tomb).
// It faces north (+y).
//
// Each layer 'k' (0-indexed from the bottom, so k=0 is the base layer) is
// represented as a single cube primitive for rendering efficiency.
//
// - The number of stones per side for layer 'k' is `num_base_stones - k`.
// - The total width/depth of layer 'k' is `(num_base_stones - k) * stone_size`.
// - Each layer is placed at a height of `k * stone_size`.
// - To center the pyramid, the base layer's bottom-left corner is at `(-base_width/2, -base_width/2, 0)`.
// - Each subsequent layer 'k' is offset inwards by `k * (stone_size / 2)` from the
//   original base layer's edges on both X and Y axes.
//   So, the bottom-left corner of layer 'k' is at `(-base_width/2 + k*stone_size/2, -base_width/2 + k*stone_size/2, k*stone_size)`.

union() {
    for (k = [0 : num_base_stones - 1]) {
        // Calculate the number of stones per side for the current layer
        current_layer_stones_per_side = num_base_stones - k;
        
        // Calculate the actual width and depth of the current layer
        layer_width_depth = current_layer_stones_per_side * stone_size;
        
        // Calculate the inward offset for the current layer to ensure it's centered
        // and steps back by stone_size/2 from the layer below on each side.
        offset_xy = k * stone_size / 2;
        
        // Translate to the bottom-left-front corner of the current layer
        // `cube(size, center=false)` creates a cube with its [0,0,0] corner at the origin.
        // We calculate the required coordinates for this corner.
        translate([ -base_width / 2 + offset_xy, 
                     -base_width / 2 + offset_xy, 
                     k * stone_size ]) {
            // Create the cube for the current layer
            cube([layer_width_depth, layer_width_depth, stone_size], center = false);
        }
    }
}
