// Pharaoh's Pyramid Project
// Units: Metric (Meters)
//
// CONSTRAINTS ANALYSIS:
// 1. Stone Budget: 150,000 stones.
// 2. Stone Size: 0.5m x 0.5m x 0.5m cube.
// 3. Support Logic: Layer N (counting from top=1) requires N*N grid of stones.
//    Total Stones = Sum(n^2) for n=1 to H.
//    Sum(n^2) = n(n+1)(2n+1)/6
//
// CALCULATIONS:
// Target <= 150,000 stones.
// Try H=76: 76(77)(153)/6 = 149,226 stones. (Fits)
// Try H=77: 77(78)(155)/6 = 155,155 stones. (Exceeds budget)
// -> Max Height in Layers = 76.
//
// DIMENSIONS:
// Total Height = 76 layers * 0.5m = 38.0m.
// Base Width (Core) = 76 stones * 0.5m = 38.0m.
//
// FACADE DESIGN:
// To create a smooth pyramid that covers the stepped core exactly:
// The plane must pass through the top-outer edge of the highest stone (Layer 1)
// and the top-outer edge of the lowest stone (Layer 76).
//
// Layer 1 Top Edge: X = 0.25m, Z = 38.0m
// Layer 76 Top Edge: X = 19.0m, Z = 0.5m
// Slope = dZ/dX = (38.0 - 0.5) / (0.25 - 19.0) = 37.5 / -18.75 = -2.
//
// Intercept (Apex Height):
// Z = -2X + C
// 38 = -2(0.25) + C => 38 = -0.5 + C => C = 38.5m.
//
// Base Width (at Z=0):
// 0 = -2X + 38.5 => 2X = 38.5 => X = 19.25m.
// Full Base Width = 38.5m.

// --- PARAMETERS ---
stone_side = 0.500; // 50cm
layers = 76;
used_stones = 149226;
core_base_width = layers * stone_side;
core_height = layers * stone_side;

// Facade Parameters calculated to perfectly touch outer edges
facade_height = core_height + (stone_side); // 38.5m
facade_base_width = core_base_width + (stone_side); // 38.5m

// --- VIEW CONTROLS ---
show_core = true;
show_facade = true;
cutaway_view = false; // Set to true to see internal stacking

module stone_core() {
    color([0.6, 0.5, 0.4]) // Sandstone color
    for (i = [1 : layers]) {
        // Optimization: Instead of rendering n*n individual cubes,
        // render one large slab per layer.
        // i = 1 is the top layer.
        layer_width = i * stone_side;
        
        // Z Position: 
        // Layer 1 sits on top of Layer 2... Layer 76 is at bottom.
        // Bottom of layer i is at Z = (total_layers - i) * size
        z_pos = (layers - i) * stone_side;
        
        translate([0, 0, z_pos + stone_side/2])
            cube([layer_width, layer_width, stone_side], center=true);
    }
}

module marble_facade() {
    // Smooth marble casing
    color([0.95, 0.95, 1.0, 0.4]) // White Marble, slightly transparent
    translate([0,0,0])
        rotate([0,0,45]) // Rotate 45 to align cylinder($fn=4) with axes
        cylinder(
            h = facade_height, 
            r1 = (facade_base_width / sqrt(2)), // Radius to corner
            r2 = 0, 
            $fn = 4
        );
}

module project_assembly() {
    difference() {
        union() {
            if (show_core) stone_core();
            if (show_facade) marble_facade();
        }
        
        // Cutaway logic for inspection
        if (cutaway_view) {
            color([1,0,0])
            translate([0, -facade_base_width, -1])
                cube([facade_base_width*2, facade_base_width*2, facade_height+2]);
        }
    }
}

// --- RENDER ---
// Note: Axis orientation: +Y is North.
// Pyramid is centered on [0,0,0].

echo(str("Total Stones Used: ", used_stones));
echo(str("Pyramid Height: ", core_height, "m (Core), ", facade_height, "m (Apex)"));
echo(str("Base Width: ", core_base_width, "m (Core), ", facade_base_width, "m (Clad)"));

project_assembly();

// Visualization of a single standard stone for reference (placed aside)
translate([25, 0, 0.25]) 
    color("red") 
    cube([0.5, 0.5, 0.5], center=true);
    
// Text label for reference
color("black")
translate([25, 0, 1])
    text("Ref. Stone (0.5m)", size=1, halign="center");
