// OpenSCAD model of the tallest possible pyramid
// using 2,500,000 stones of 500mm cubes.
// Uses exactly 2,490,670 stones for maximum height of 195 layers.
// Each layer is a solid block equivalent to k x k stones, k=195 (bottom) to 1 (top).
// Centered at origin, base at z=0, height +z.
// North is +y direction.
// Units: mm (metric, precise to 1mm).
// Optimized: only 195 primitives, viewable instantly.

stone_side = 500;  // mm
num_layers = 195;  // Maximum layers without exceeding stone budget

pyramid();

module pyramid() {
    for (i = [0 : num_layers - 1]) {
        side_stones = num_layers - i;
        layer_size = side_stones * stone_side;  // mm
        layer_bottom_z = i * stone_side;        // mm
        translate([0, 0, layer_bottom_z + stone_side / 2])
            cube([layer_size, layer_size, stone_side], center = true);
    }
}
