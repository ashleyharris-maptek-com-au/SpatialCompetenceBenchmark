            // Multiply by `stone_size` to get the spacing.
            x_local_center = (x_idx - (stones_on_side - 1) / 2.0) * stone_size;
            y_local_center = (y_idx - (stones_on_side - 1) / 2.0) * stone_size;

            // Apply the layer offset shift.
            // This shift accounts for the stepping of the pyramid.
            // The shift is half a stone size in x and y for each subsequent layer.
            global_x_center = x_local_center + offset_shift * stone_size;
            global_y_center = y_local_center + offset_shift * stone_size;

            // The z coordinate is the bottom of the layer.
            // We need to place the stone centered at (global_x_center, global_y_center, z_pos + stone_size/2).
            // The cube module's origin is at its minimum corner.
            // To place a cube of size S centered at (cx, cy, cz), we translate by (cx - S/2, cy - S/2, cz - S/2).
            translate([global_x_center - stone_size/2, global_y_center - stone_size/2, z_pos]) {
                stone();
            }
        }
    }
}

// Module for a single stone (50cm cube)
module stone() {
    cube([stone_size, stone_size, stone_size]);
}
