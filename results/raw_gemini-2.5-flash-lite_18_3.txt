```openscad
// Pharaoh's Pyramid Plan
// Units: Millimeters (mm)

// --- Constants ---
// Stone dimensions in mm (50cm = 500mm)
STONE_SIDE = 500;
// Stone volume in mm^3
STONE_VOLUME = STONE_SIDE * STONE_SIDE * STONE_SIDE;

// --- Pharaoh's Resources ---
// Total available stones
MAX_STONES = 2500000;
// Men required to move one stone
MEN_PER_STONE = 10;
// Total men available
TOTAL_MEN = 100000;
// Maximum number of stones that can be moved at any given time.
// This is limited by the number of men and how many are needed per stone.
// This constraint implies that the construction must be sequential, layer by layer,
// and the total number of men available might limit the rate of construction,
// but not the ultimate size of the pyramid if enough time is allocated.
// The problem statement focuses on the budget of stones.
// Therefore, we assume time is not a limiting factor for the number of stones.

// --- Pyramid Geometry Rules ---
// Each stone's base must have all 4 corners sitting in the middle of another stone.
// This means each layer above the base must be smaller than the layer below.
// For a square base, this implies a step down in size.
// If a stone is of size S, and its corners are at (x,y), (x+S,y), (x,y+S), (x+S,y+S),
// then the center of the stone below it must be at the corner.
// This leads to a pattern where each subsequent layer's stones are placed
// on top of stones in the layer below, offset by half a stone's width.
// This effectively means that each layer will have a side length reduction.
// A layer of size N stones by N stones will sit on a layer of size N+1 stones by N+1 stones.
// The offset means that the 'footprint' of a layer is effectively one stone less
// in each dimension than the layer it sits upon.
//
// Base layer: N x N stones
// Layer 1: (N-1) x (N-1) stones
// Layer 2: (N-2) x (N-2) stones
// ...
// Top layer: 1 x 1 stone
//
// Total stones = N^2 + (N-1)^2 + ... + 1^2

// --- Calculation of Tallest Pyramid ---
// We need to find the largest N such that the sum of squares from 1 to N is <= MAX_STONES.
// Sum of squares formula: N(N+1)(2N+1) / 6

// Let's iterate to find the maximum N.
// We can start with an estimate for N and refine it.
// The volume is roughly N^3 / 3. So N is roughly cube root of (3 * MAX_STONES).
// 3 * 2500000 = 7500000
// cube root of 7500000 is about 195.

max_n = 0;
current_stones_used = 0;
for (n = 1; n <= 300; n++) { // Iterate up to a reasonable limit
    stones_in_layer_n = n * n;
    stones_in_layer_n_minus_1 = (n - 1) * (n - 1);
    // Calculate total stones for a pyramid of height n (meaning base is n x n)
    // The sum of squares formula calculates the number of stones for a pyramid
    // with base size N x N. So, a pyramid of height N has N layers.
    // The total number of stones for a pyramid of height H (base HxH) is sum(i^2 for i from 1 to H)
    total_stones_for_height_n = n * (n + 1) * (2 * n + 1) / 6;

    if (total_stones_for_height_n <= MAX_STONES) {
        max_n = n;
        current_stones_used = total_stones_for_height_n;
    } else {
        break; // We've exceeded the stone budget
    }
}

// max_n is the side length of the base layer in terms of stones.
// The height of the pyramid will be max_n layers.

PYRAMID_BASE_SIDE_STONES = max_n;
PYRAMID_HEIGHT_LAYERS = max_n;
TOTAL_STONES_USED = current_stones_used;

// --- OpenSCAD Module for a Single Stone ---
module stone() {
    cube([STONE_SIDE, STONE_SIDE, STONE_SIDE]);
}

// --- OpenSCAD Module for the Pyramid ---
module pharaohs_pyramid(base_side_stones, height_layers) {
    // The pyramid is centered over the origin (0,0,0).
    // The base layer will be at z=0.
    // For orientation, +y is North.

    // Calculate the total dimensions of the pyramid's base for centering
    base_layer_width_mm = base_side_stones * STONE_SIDE;
    base_layer_depth_mm = base_side_stones * STONE_SIDE;

    // Offset to center the base layer
    offset_x = -base_layer_width_mm / 2;
    offset_y = -base_layer_depth_mm / 2;
    offset_z = 0; // Base starts at z=0

    for (layer = 0; layer < height_layers; layer++) {
        // Current layer number (0-indexed from the bottom)
        current_layer_idx = layer;
        // Number of stones along the side of the current layer
        // Layer 0 (base): PYRAMID_BASE_SIDE_STONES stones
        // Layer 1: PYRAMID_BASE_SIDE_STONES - 1 stones
        // Layer k: PYRAMID_BASE_SIDE_STONES - k stones
        stones_on_side = base_side_stones - current_layer_idx;

        if (stones_on_side <= 0) break; // Should not happen with correct max_n calculation

        layer_width_mm = stones_on_side * STONE_SIDE;
        layer_depth_mm = stones_on_side * STONE_SIDE;
        layer_height_mm = (current_layer_idx + 1) * STONE_SIDE; // Height accumulated from base

        // Calculate offset for this layer to center it on the layer below
        // The rule implies each layer is offset by half a stone in x and y.
        // For a layer of N x N stones, it sits on an (N+1) x (N+1) layer.
        // The offset of the N x N layer relative to the (N+1) x (N+1) layer center is 0.5 stone.
        // So, a layer of 'stones_on_side' will be offset by (PYRAMID_BASE_SIDE_STONES - stones_on_side) * 0.5 * STONE_SIDE
        // from the base layer's center.
        // Each layer is centered relative to the layer below.
        // The difference in side length between consecutive layers is 1 stone.
        // For layer `l` (0-indexed from base), it has `base_side_stones - l` stones.
        // The layer below it has `base_side_stones - (l-1)` stones.
        // The offset required to center layer `l` on layer `l-1` is:
        // (layer_below_stones_on_side - layer_this_stones_on_side) * 0.5 * STONE_SIDE
        // which simplifies to 0.5 * STONE_SIDE.

        // The centering offset for layer `l` relative to the absolute center (origin) is
        // the sum of offsets from the base.
        // Offset for layer `l` = (base_side_stones - (base_side_stones - l)) * 0.5 * STONE_SIDE = l * 0.5 * STONE_SIDE
        // This offset should be applied to the start of the layer's bounding box.
        // The starting x coordinate for layer `l` will be:
        // offset_x (base centering) + l * STONE_SIDE / 2
        // Similarly for y.

        layer_offset_x = offset_x + current_layer_idx * STONE_SIDE / 2;
        layer_offset_y = offset_y + current_layer_idx * STONE_SIDE / 2;
        layer_offset_z = current_layer_idx * STONE_SIDE;

        for (i = 0; i < stones_on_side; i++) {
            for (j = 0; j < stones_on_side; j++) {
                // Position of the bottom-left corner of the current stone in this layer
                stone_pos_x = layer_offset_x + i * STONE_SIDE;
                stone_pos_y = layer_offset_y + j * STONE_SIDE;
                stone_pos_z = layer_offset_z;

                translate([stone_pos_x, stone_pos_y, stone_pos_z]) {
                    stone();
                }
            }
        }
    }
}

// --- Pyramid Properties ---
// Calculate the exact total volume of stones used.
// Using the sum of squares formula gives the count of stones.
// Total volume = TOTAL_STONES_USED * STONE_VOLUME
TOTAL_VOLUME_MM3 = TOTAL_STONES_USED * STONE_VOLUME;

// Calculate the height of the pyramid in mm.
// The height is the number of layers multiplied by the height of each stone.
PYRAMID_HEIGHT_MM = PYRAMID_HEIGHT_LAYERS * STONE_SIDE;

// The pyramid is centered over the origin (0,0,0).
// +y is North.
pharaohs_pyramid(PYRAMID_BASE_SIDE_STONES, PYRAMID_HEIGHT_LAYERS);
```
