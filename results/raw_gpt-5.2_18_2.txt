// Tallest stable stepped pyramid given 600,000 cubes of 0.5 m side.
// Coordinate system: +Y = North. Pyramid centered over tomb at (0,0).
// Units: millimeters.

$fn = 24;

// --- Parameters ---
stone_mm = 500;                 // 50 cm cube
available_stones = 600000;

// Stability rule implies each upper stone's corners must lie within the central area of a lower stone.
// With equal cubes aligned to axes, this requires each higher layer to be inset by at least 250 mm
// from each edge relative to the layer below => side length decreases by exactly 500 mm per layer.
step_inset_mm = stone_mm / 2;    // 250 mm per side
side_shrink_per_layer_mm = 2 * step_inset_mm; // 500 mm total shrink per layer (both sides)

// Choose integer number of stones per side on base layer: N0.
// Layer k has Nk = N0 - k stones per side (k = 0..H-1), so H = N0 layers.
// Total stones S(N0) = sum_{n=1..N0} n^2 = N0(N0+1)(2N0+1)/6.
// Max N0 with S(N0) <= 600,000 is N0 = 121 (S=597,861). N0=122 would exceed.
N0 = 121;
H  = N0;

used_stones = N0*(N0+1)*(2*N0+1)/6;

// --- Geometry helpers ---
module stepped_pyramid(N_base)
{
    // Build as union of layer slabs for efficiency:
    // Each layer is a rectangular prism of height 500 mm, side = Nk*500 mm.
    // Centered at origin, aligned to axes; +Y is North.
    union() {
        for (k = [0 : N_base-1]) {
            Nk = N_base - k;
            side = Nk * stone_mm;
            // Layer k sits from z = k*500 to (k+1)*500
            translate([0, 0, k*stone_mm])
                cube([side, side, stone_mm], center=true);
        }
    }
}

// --- Output ---
/*
Summary (for reference; keep as comments only):
Base stones per side: 121
Height layers:        121
Stone used:           597,861 (<= 600,000)
Unused stones:        2,139
Base side length:     60,500 mm
Total height:         60,500 mm
*/
stepped_pyramid(N0);
