// Pyramid specifications
// Stone size: 50cm (0.5m) cubes
// Placement rule: corners at centers of stones below
// This creates offset of 25cm (0.25m) per layer
// Available stones: 600,000

stone_size = 500; // mm
offset = 250; // mm (half stone size)

// Calculate maximum pyramid
// Layer n has side length: base_side - 2*(n-1)*offset
// Stones in layer n: ((base_side - 2*(n-1)*offset)/stone_size)^2
// Total stones for h layers: sum of all layer stones <= 600,000

// For pyramid with base side b stones:
// Total = sum(i^2) for i=1 to b = b(b+1)(2b+1)/6
// Solving b(b+1)(2b+1)/6 <= 600000
// b^3 ≈ 1,800,000, b ≈ 121.6
// Testing: b=121: 121*122*243/6 = 596,839 stones
// b=122: 122*123*245/6 = 613,970 stones (too many)

base_stones = 121;
base_side = base_stones * stone_size;
layers = base_stones;
pyramid_height = layers * stone_size;

// Create smooth pyramid hull using corner points
module smooth_pyramid() {
    hull() {
        // Base corners
        translate([-base_side/2, -base_side/2, 0]) cube(1);
        translate([base_side/2, -base_side/2, 0]) cube(1);
        translate([-base_side/2, base_side/2, 0]) cube(1);
        translate([base_side/2, base_side/2, 0]) cube(1);
        
        // Apex
        translate([0, 0, pyramid_height]) cube(1);
    }
}

smooth_pyramid();
