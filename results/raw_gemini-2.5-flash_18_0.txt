```openscad
// Pharaoh's Pyramid Plan
// This OpenSCAD script generates the precise layout of stones for the tallest pyramid
// the pharaoh can build according to the specified constraints.

// --- Parameters ---
cube_size = 0.5; // Side length of each large stone cube, in meters (50cm)
num_layers = 38; // The total number of layers for the pyramid (calculated to use max stones)
                 // Calculated as the largest N such that Sum(k^2 for k from 1 to N) <= 20,000
                 // For N=38, total stones = 38*(38+1)*(2*38+1)/6 = 18,989 stones.
                 // The next layer (N=39) would require 39^2 = 1,521 stones, exceeding the budget.

// --- Pyramid Construction ---
// The pyramid is centered over the future tomb (origin 0,0) and faces North (+Y).
// The base of the pyramid rests on the ground (Z=0).
// Each layer 'k' (from 0 for the base to num_layers-1 for the top)
// has (num_layers - k) x (num_layers - k) stones.
// Stones in layer 'k' are offset by (k * cube_size / 2) in X and Y relative to the
// overall alignment of the base layer, ensuring that each stone's 4 base corners
// sit on the centers of 4 stones in the layer directly below it.

for (k = 0; k < num_layers; k = k + 1) {
    // Calculate the number of stones along one side for the current layer
    current_layer_stones_side = num_layers - k;

    // Calculate the base offset for centering the entire pyramid at (0,0)
    // The total width of the base layer is num_layers * cube_size.
    // We want the center of this base to be at X=0, Y=0.
    // The bottom-left corner of the layer 0 (base) would start at
    // - (num_layers * cube_size / 2)
    // Then we add half a cube_size to place the center of the first cube
    // And then (k * cube_size / 2) for the staggering offset of the current layer.
    base_offset_x = - (num_layers * cube_size / 2) + (cube_size / 2) + (k * cube_size / 2);
    base_offset_y = - (num_layers * cube_size / 2) + (cube_size / 2) + (k * cube_size / 2);

    for (j = 0; j < current_layer_stones_side; j = j + 1) {
        for (i = 0; i < current_layer_stones_side; i = i + 1) {
            // Calculate the center coordinates for the current stone
            // X-coordinate: base_offset_x + (index_i * cube_size)
            // Y-coordinate: base_offset_y + (index_j * cube_size)
            // Z-coordinate: (layer_k * cube_size) + (cube_size / 2) (for center=true)
            translate([
                base_offset_x + (i * cube_size),
                base_offset_y + (j * cube_size),
                (k * cube_size) + (cube_size / 2)
            ]) {
                cube(cube_size, center = true);
            }
        }
    }
}
```
