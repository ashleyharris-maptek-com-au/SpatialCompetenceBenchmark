// Tallest step pyramid under given constraints
// Units: meters (stone_size = 0.5 m = 500 mm)

stone_size = 0.5;
total_stones_available = 150000;

// Derived by solving n(n+1)(2n+1)/6 <= 150000
layers = 76; // number of vertical stone layers
stones_used = layers * (layers + 1) * (2 * layers + 1) / 6;

// Pyramid overall dimensions
base_side = layers * stone_size;
height    = layers * stone_size;

// Core step pyramid made of solid square layers (no internal voids)
module step_pyramid() {
    union() {
        for (level = [0 : layers - 1]) {
            layer_cubes = layers - level;
            layer_side  = layer_cubes * stone_size;
            translate([0, 0, stone_size * (level + 0.5)])  // center each layer
                cube([layer_side, layer_side, stone_size], center = true);
        }
    }
}

// Smooth marble envelope (right square pyramid)
// One triangular face is oriented toward +y (north).
module smooth_pyramid() {
    points = [
        [-base_side/2, -base_side/2, 0],  // 0: SW base corner
        [ base_side/2, -base_side/2, 0],  // 1: SE base corner
        [ base_side/2,  base_side/2, 0],  // 2: NE base corner (north edge)
        [-base_side/2,  base_side/2, 0],  // 3: NW base corner
        [ 0,            0,           height] // 4: apex
    ];

    // Triangulated faces, oriented so normals point outward
    faces = [
        [0, 3, 2],  // base, triangle 1
        [0, 2, 1],  // base, triangle 2
        [0, 1, 4],  // south face  (normal ~ -y)
        [1, 2, 4],  // east  face  (normal ~ +x)
        [2, 3, 4],  // north face  (normal ~ +y)
        [3, 0, 4]   // west  face  (normal ~ -x)
    ];

    polyhedron(points = points, faces = faces);
}

// Marble cladding volume: space between smooth envelope and stone core.
// (This yields the wedges needed to cover every external stone.)
module marble_cladding() {
    difference() {
        smooth_pyramid();
        step_pyramid();
    }
}

// - Stone core in a sandstone color.
// - Optional smooth envelope or cladding can be enabled via the comments.

color([0.8, 0.7, 0.5]) step_pyramid();
