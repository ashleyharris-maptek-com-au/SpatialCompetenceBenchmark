// Pharaoh's Pyramid Plan
// Tallest pyramid achievable with manpower constraint:
// - Base: 30 x 30 stones (L = 30), each stone = 50 cm (= 500 mm) cube
// - Layering rule: each upper stone rests on four stones below (2x2 grid)
 // -> total stones used = sum_{k=1}^{30} k^2 = 9,455
// - Stones available: 150,000; manpower: 100,000; 10 men per stone -> max movable stones = 10,000
// - 9,455 <= 10,000, so a 30-layer pyramid fits; height = 30 * 0.5 m = 15 m
// - Plan is centered so the apex lies above the tomb center (origin) with north (+y) facing forward.

L = 30;            // number of stones along base per side
block = 500;         // mm side length per stone (50 cm)
base_size = L * block; // total base dimension in mm (15000 mm)
half_base = base_size / 2; // 7500 mm

// Stone rendering: outer appearance per layer (optional color coding for clarity).
// Top layer (z = L-1) will be colored red to show apex.
// Other layers colored in a gentle gray for readability.

module stone_cube() {
    cube([block, block, block], center = false);
}

// The pyramid is built such that:
 // For each layer z (0-based from bottom), there are (L - z) x (L - z) stones.
 // The x,y origin of a stone at (i,j,z) is:
//   x0 = i*block + z*(block/2) - half_base
//   y0 = j*block + z*(block/2) - half_base
 // This centers the entire pyramid over (0,0) so apex is above the tomb center.
for (z = [0 : L - 1]) {
    dim = L - z;
    for (i = [0 : dim - 1]) {
        for (j = [0 : dim - 1]) {
            x0 = i * block + z * (block / 2) - half_base;
            y0 = j * block + z * (block / 2) - half_base;
            // Position of the stone's bottom-left-front corner
            pos = [x0, y0, z * block];
            translate(pos) {
                if (z == L - 1) {
                    color([1.0, 0.0, 0.0])
                    stone_cube();
                } else {
                    color([0.80, 0.82, 0.92])
                    stone_cube();
                }
            }
        }
    }
}
