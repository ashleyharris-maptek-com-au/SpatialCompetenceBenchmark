// OpenSCAD Plan for Pharaoh's Pyramid
// =====================================
// UNITS: Meters (1 unit = 1 meter)
//
// DESIGN CONSTRAINTS:
// 1. Resource Limit: 600,000 stones (50cm cubes).
// 2. Structural Rule: 50% offset stacking (corners on centers).
//    - This implies layer n (from top) has n*n stones.
//    - Total stones = Sum(k^2) for k=1 to Height.
// 3. Optimization: Maximize Height (H).
//    - Sum(k^2, 1, 121) = 597,903 stones.
//    - Sum(k^2, 1, 122) = 612,745 stones (Exceeds budget).
//    - MAX LAYERS = 121.
// 4. Orientation: Faces North (+Y).
// 5. Facade: Smooth marble wedges covering the steps.

// --- CONFIGURATION ---

// Dimensions in Meters
stone_size = 0.500; 
num_layers = 121;   
total_stones_used = 597903; 

// The core is a stepped pyramid.
// Base width (Layer 121) = 121 * 0.5 = 60.5m
// Top width (Layer 1)    = 1 * 0.5 = 0.5m
// Total Stone Height     = 121 * 0.5 = 60.5m

// The smooth facade is defined by the outer corners of the steps.
// Bottom-most Stone Corner: x = 30.25m, z = 0.5m (Top of bottom block)
// Next Stone Corner:        x = 30.00m, z = 1.0m
// Slope (dz/dx) = (1.0 - 0.5) / (30.00 - 30.25) = -2.
//
// Projected Facade Geometry:
// Line equation: z = -2x + C
// Fitting (30.25, 0.5): 0.5 = -2(30.25) + C => 0.5 = -60.5 + C => C = 61.
// Apex Height (x=0) = 61.0m.
// Base Width (z=0)  => 0 = -2x + 61 => x = 30.5m => Width = 61.0m.

facade_base_width = 61.0;
facade_height = 61.0;

// --- VOLUME CALCULATIONS (Echo to Console) ---
echo("-----------------------------------------------------");
echo("PHARAOH'S PYRAMID CONSTRUCTION PLAN");
echo("-----------------------------------------------------");
echo("Max Layers Possible:", num_layers);
echo("Total Stones Required:", total_stones_used);
echo("Stones Remaining:", 600000 - total_stones_used);
echo("Pyramid Base Width (Marble):", facade_base_width, "m");
echo("Pyramid Height (Apex):", facade_height, "m");
echo("Core Stone Volume (m3):", total_stones_used * pow(stone_size, 3));
// Ideal Pyramid Volume = (Base_Area * Height) / 3
// Base Area = 61 * 61 = 3721 m2
echo("Total Ideal Volume (m3):", (facade_base_width * facade_base_width * facade_height) / 3);
echo("Marble Volume Required (m3):", ((facade_base_width * facade_base_width * facade_height) / 3) - (total_stones_used * pow(stone_size, 3)));
echo("-----------------------------------------------------");

// --- RENDERING ---

// 1. The Limestone Core
// Optimized: Rendered as 121 slabs rather than 600,000 cubes.
module core_structure() {
    color([0.8, 0.75, 0.6, 1.0]) // Sandstone color
    for (i = [1 : num_layers]) {
        // Layer i (1 is top, 121 is bottom)
        // Width calculation:
        // Top layer (i=1) has width 1 * 0.5
        // Bottom layer (i=121) has width 121 * 0.5
        layer_width = i * stone_size;
        
        // Z Position:
        // Layer 1 is at top. Z = (num_layers - 1) * 0.5 ??
        // Let's stack from bottom up to match Z-axis logic.
        // Inverse index k where k=0 is bottom layer.
        k = num_layers - i; // 120 (bottom) to 0 (top)
        
        translate([0, 0, k * stone_size + (stone_size/2)])
        cube([layer_width, layer_width, stone_size], center=true);
    }
}

// 2. The Marble Facade
// Defined as the Ideal Pyramid minus the Core Structure.
module marble_facade() {
    color([0.95, 0.95, 1.0, 0.4]) // White Marble, Semi-transparent
    difference() {
        // Ideal Pyramid
        // Rotated 45 degrees so faces point N/S/E/W
        rotate([0, 0, 45])
        cylinder(r1 = (facade_base_width / 2) * sqrt(2), r2 = 0, h = facade_height, $fn=4);
        
        // Remove Core to leave only the marble wedges
        core_structure();
    }
}

// 3. Cutaway Plan View
// A boolean cut to show the internal arrangement for the stonemason.
module construction_view() {
    difference() {
        union() {
            core_structure();
            marble_facade();
        }
        // Cut out the Front-Right quadrant to show cross-section
        translate([0, -facade_base_width/2, -1])
        cube([facade_base_width, facade_base_width, facade_height + 2]);
    }
}

// --- MAIN OUTPUT ---

// Render the full assembly
// For the stonemason's CAD, we separate the parts but show them assembled.
union() {
    core_structure();
    marble_facade();
}
