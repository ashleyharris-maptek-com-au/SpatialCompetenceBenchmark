// Pyramid for the pharoh: tallest possible using 20,000 stones (50 cm cubes).
// Each stone: 500 mm side length.
// Maximum full square-layer pyramid uses N = 38 layers (sum_{k=1..38} k^2 = 19,019 stones).
// Height = 38 * 500 mm = 19,000 mm (19.0 m).
// Pyramid is centered at the origin [0,0,0]. +Y is North (face of pyramid points +Y).
// Units: millimetres (mm). All coordinates and sizes exact to the millimetre.

stone = 500;     // stone side length in mm
N = 38;          // number of layers (largest integer with sum_{k=1..N} k^2 <= 20000)

module stone_cube() {
    // centered cube so translate uses center coordinates directly
    cube([stone, stone, stone], center=true);
}

union() {
    // Build layers from base (L=0) to top (L=N-1).
    // Layer L contains S = N - L stones per side, centered at origin.
    // Stone centers on layer L lie at z = (L + 0.5)*stone.
    for (L = [0 : N-1]) {
        S = N - L;
        zc = (L + 0.5) * stone;
        // x and y coordinates: (i - (S-1)/2) * stone  (centers the layer at origin)
        for (i = [0 : S-1]) {
            for (j = [0 : S-1]) {
                xc = (i - (S-1)/2) * stone;
                yc = (j - (S-1)/2) * stone;
                translate([xc, yc, zc])
                    stone_cube();
            }
        }
    }
}
