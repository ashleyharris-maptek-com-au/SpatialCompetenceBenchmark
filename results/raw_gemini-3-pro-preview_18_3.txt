// OpenSCAD Plan for the Pharaoh's Pyramid
// -----------------------------------------------------------------------------
// Parameters and Constraints:
// Unit System: Metric (mm)
// Stone Size: 50cm x 50cm x 50cm cube (500mm)
// Stone Supply: 2,500,000 stones
// Labor: 100,000 men (Contextual constraint satisfied by stone limit)
// Orientation: Faces North (+y)
// Structure: No internal voids. 
// Stacking: 1-to-4 ratio (Corner over center) -> Sum of squares series.
// Facade: Marble wedges covering the steps to form a smooth pyramid.

// -----------------------------------------------------------------------------
// 1. Calculations
// -----------------------------------------------------------------------------

/*
   Stone Count Formula for N layers:
   Sum(n^2) = n(n + 1)(2n + 1) / 6
   
   Solving for Sum <= 2,500,000:
   N = 195
   Stones Used = 195 * 196 * 391 / 6 = 2,490,670 stones
   (Next layer 196 requires 38,416 stones, exceeding budget)
*/

stone_size_mm = 500; 
num_layers = 195;
stones_used = 2490670;

// Dimensions of the Core (Stepped)
core_base_width = num_layers * stone_size_mm; // 97,500 mm
core_height = num_layers * stone_size_mm;     // 97,500 mm

// Dimensions of the Facade (Smooth)
// To cover the stepped core where slope = 2 (Rise 1 unit per 0.5 unit offset),
// and touching the top-outer edge of every stone to minimize marble usage:
// The geometric apex must be one stone-height above the top stone's top face.
// The geometric base must be 0.5 stone-widths wider on all sides than the bottom layer.

facade_apex_height = (num_layers + 1) * stone_size_mm; // 98,000 mm
facade_base_width  = (num_layers + 1) * stone_size_mm; // 98,000 mm

// Output Stats for the Stonemason
echo("-------------------------------------------------------");
echo("Pharaoh's Pyramid Plan");
echo("-------------------------------------------------------");
echo("Total Layers:", num_layers);
echo("Total Height (m):", facade_apex_height / 1000);
echo("Base Width (m):", facade_base_width / 1000);
echo("Core Stones Used:", stones_used);
echo("Stones Remaining:", 2500000 - stones_used);
echo("Marble Volume approx (m3):", ((1/3) * pow(facade_base_width/1000, 2) * (facade_apex_height/1000)) - (stones_used * pow(stone_size_mm/1000, 3)));
echo("-------------------------------------------------------");

// -----------------------------------------------------------------------------
// 2. Geometry Modules
// -----------------------------------------------------------------------------

// The inner structural core made of sandstone cubes
module core_structure() {
    color("SandyBrown") 
    union() {
        for (i = [1 : num_layers]) {
            // Layer 1 is top, Layer N is bottom
            // Width increases linearly
            layer_w = i * stone_size_mm;
            
            // Calculate Elevation (Bottom of the pyramid is Z=0)
            // Layer i sits on top of Layer i+1.
            // Bottom layer (195) is at Z=0.
            elevation = (num_layers - i) * stone_size_mm;
            
            translate([0, 0, elevation + stone_size_mm/2])
                cube([layer_w, layer_w, stone_size_mm], center=true);
        }
    }
}

// The smooth outer shape defined by the casing
module smooth_envelope() {
    // A pyramid with 4 sides.
    // Cylinder with $fn=4 creates a prism.
    // By default, vertices align with X/Y axes at 0, 90, 180, 270.
    // Flat faces are at 45 degrees.
    // To have faces pointing North (+Y), East, South, West, we rotate 45 degrees.
    
    // Radius Calculation:
    // For a square base of width W, the circumradius R (distance to corner) is W / sqrt(2).
    R = facade_base_width / sqrt(2);
    
    color("White")
    rotate([0, 0, 45])
        cylinder(r1 = R, r2 = 0, h = facade_apex_height, $fn=4);
}

// The marble casing (Difference between smooth envelope and core)
module marble_facade() {
    color("White", 0.9)
    difference() {
        smooth_envelope();
        // We extend the core slightly for the difference operation to avoid Z-fighting
        // on the touching edges during rendering, though mathematically exact is 
        // preferred for plans. We use the exact core here for precision.
        core_structure();
    }
}

// -----------------------------------------------------------------------------
// 3. Assembly / Rendering
// -----------------------------------------------------------------------------

// To help the stonemason visualize, we render the core and the facade together.
// The transparency in the marble_facade color helps visualize the covering.

union() {
    core_structure();
    marble_facade();
}
