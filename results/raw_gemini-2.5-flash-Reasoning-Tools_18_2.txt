```openscad
// Define stone dimensions and pyramid parameters
stone_s = 500; // Side length of each cubic stone in millimeters
N_base = 121;  // Number of stones on one side of the base layer.
               // This value is calculated to maximize height given 600,000 stones
               // N=121 uses 597,861 stones (sum of squares from 1 to 121).

// Module to define a single stone.
// Using center=true makes its origin the center of the cube, which simplifies placement.
module stone() {
    cube(stone_s, center=true);
}

// Loop through each layer of the pyramid, starting from the base (k=0) to the top (k=N_base-1).
// Each increment of 'k' represents moving one layer up the pyramid.
for (k = [0 : N_base - 1]) {
    // Calculate the number of stones on one side for the current layer k.
    // The base layer (k=0) has N_base stones.
    // Each subsequent layer (k=1, k=2, ...) has one fewer stone per side: N_base-1, N_base-2, etc.
    // The top layer (k=N_base-1) will have 1 stone per side (N_base - (N_base-1) = 1).
    num_stones_side_k = N_base - k;

    // Translate the entire current layer upwards by 'k * stone_s'.
    // This stacks the layers vertically on top of each other.
    // The XY translation is 0 because each layer is independently centered by the inner loops.
    translate([0, 0, k * stone_s]) {
        // Loop through the X-axis positions for stones within the current layer.
        for (i = [0 : num_stones_side_k - 1]) {
            // Loop through the Y-axis positions for stones within the current layer.
            for (j = [0 : num_stones_side_k - 1]) {
                // Calculate the X and Y coordinates for the center of each stone within its layer.
                // - `i * stone_s` and `j * stone_s` give the coordinates of the bottom-left corner
                //   of a stone if they were arranged starting from (0,0) without centering.
                // - `+ (stone_s / 2)` shifts the reference point from the corner to the stone's center.
                // - `- (num_stones_side_k * stone_s / 2)` then shifts the entire grid of stones
                //   for this layer so that its overall center aligns with (0,0) in the XY plane.
                //   This ensures the pyramid is centered over the tomb.
                // The Z-translation is 0 here because the layer's base height is already handled
                // by the outer `translate([0, 0, k * stone_s])`.
                translate([i * stone_s + (stone_s / 2) - (num_stones_side_k * stone_s / 2),
                           j * stone_s + (stone_s / 2) - (num_stones_side_k * stone_s / 2),
                           0])
                {
                    stone(); // Place the stone cube at the calculated position.
                }
            }
        }
    }
}
```
