// Pharaoh's Pyramid Core and Marble Casing
// Units: meters (SI)

stone_size = 0.5;       // Edge length of each cubic core stone (0.5 m)
layers     = 38;        // Number of stone layers (max with 20,000 stones)
base_count = layers;    // Stones per side in the bottom layer
half_base  = base_count * stone_size / 2;  // Half side length of core base (9.5 m)

base_z     = stone_size;                 // Top of bottom stone layer (0.5 m)
apex_z     = (layers + 1) * stone_size;  // Apex height of marble casing (19.5 m)

show_core   = true;   // Set to false to hide the stone core
show_casing = true;   // Set to false to hide the marble casing


// Main assembly
if (show_core)
    color([0.8, 0.7, 0.5]) core_pyramid();       // Sandstone-like color

if (show_casing)
    color([1, 1, 1, 0.5]) marble_casing();       // Semi-transparent white casing


// Core pyramid made of individual cubic stones
module core_pyramid() {
    union() {
        for (k = [0 : layers - 1]) {                       // Layer index (0 = bottom)
            for (i = [0 : base_count - 1 - k])            // Stones in X for this layer
                for (j = [0 : base_count - 1 - k]) {      // Stones in Y for this layer
                    translate([
                        -half_base + (stone_size/2)*k + i*stone_size,
                        -half_base + (stone_size/2)*k + j*stone_size,
                        k*stone_size
                    ])
                        cube(stone_size, center = false);
                }
        }
    }
}


// Smooth marble casing pyramid
// Faces are aligned so that all exposed top outer corners of the stone core
// lie exactly on the casing surface, giving a smooth outer pyramid.
module marble_casing() {
    polyhedron(
        points = [
            [ half_base,  half_base,  base_z],  // 0: NE base corner  (x>0, y>0)
            [-half_base,  half_base,  base_z],  // 1: NW base corner  (x<0, y>0)
            [-half_base, -half_base,  base_z],  // 2: SW base corner  (x<0, y<0)
            [ half_base, -half_base,  base_z],  // 3: SE base corner  (x>0, y<0)
            [ 0,          0,          apex_z]   // 4: Apex (above tomb center)
        ],
        faces = [
            // Side faces (counter-clockwise as seen from outside)
            [0, 1, 4],  // North face  (normal points toward +y)
            [1, 2, 4],  // West face   (normal points toward -x)
            [2, 3, 4],  // South face  (normal points toward -y)
            [3, 0, 4],  // East face   (normal points toward +x)

            // Base (oriented downward)
            [0, 3, 2],
            [0, 2, 1]
        ]
    );
}
