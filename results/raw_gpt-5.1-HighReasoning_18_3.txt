// Tallest pyramid core and marble casing for 2,500,000 stones
// Units: millimetres (SI)

block_size_mm = 500;             // Edge length of each stone (50 cm)
stones_available = 2500000;
base_blocks = 195;               // Stones per side at base (max that fits budget)
layers = base_blocks;            // One layer per stone in height

core_height = layers * block_size_mm;
core_half = base_blocks * block_size_mm / 2;  // 48,750 mm

// Smooth marble casing chosen to just cover the core:
// base half-size 49,000 mm, height 98,000 mm (slope = 2:1)
outer_half = 49000;
outer_height = 98000;

// Stones actually used in the core pyramid
stones_used = base_blocks * (base_blocks + 1) * (2 * base_blocks + 1) / 6;
assert(stones_used <= stones_available);

// Step-pyramid stone core (no internal voids)
module core_pyramid() {
    for (layer = [0 : layers - 1]) {
        n = base_blocks - layer;                     // Stones per side in this layer
        side = n * block_size_mm;                    // Layer side length
        translate([0, 0, block_size_mm/2 + layer * block_size_mm])
            cube([side, side, block_size_mm], center = true);
    }
}

// Smooth marble casing as a square pyramid facing +Y (north)
module marble_casing() {
    linear_extrude(height = outer_height, scale = 0)
        square([2 * outer_half, 2 * outer_half], center = true);
}

// View options:

// 1) Core only:
// color([0.8, 0.7, 0.5]) core_pyramid();

// 2) Marble casing only:
// color([1, 1, 1]) marble_casing();

// 3) Core with translucent marble casing:
color([0.8, 0.7, 0.5]) core_pyramid();
color([1, 1, 1, 0.4]) marble_casing();
