```openscad
// OpenSCAD Plan for Pharaoh's Pyramid
// ===================================
// Units: Meters (SI)
//
// CALCULATION LOG:
// ----------------
// Stone Size (cube): 0.5m x 0.5m x 0.5m
// Available Men: 100,000
// Men required per stone: 10
// Max Moveable Stones (Manpower Limit): 100,000 / 10 = 10,000 stones
// Available Stones (Inventory): 20,000
// Limiting Factor: Manpower (10,000 stones max)
//
// Pyramid Structure:
// Layer n (from top) is an n x n grid of stones.
// Total Stones = Sum(i^2) for i = 1 to H
//
// Height Iteration:
// H=30: Sum(1^2...30^2) = (30*31*61)/6 = 9,455 stones (Fits)
// H=31: Sum(1^2...31^2) = 10,416 stones (Exceeds limit)
//
// Result: 
// Max Height: 30 layers = 15.0 meters.
// Base Width: 30 * 0.5m = 15.0 meters.
// Total Stones Used: 9,455
//
// Facade Geometry:
// The marble facade must cover the steps to form a smooth pyramid.
// To cover the top-most stone (Width 0.5m at Z=15m) and the bottom-most 
// stone corner (Width 15m at Z=0m), the smooth apex must be higher than the stone core.
//
// Apex Calculation:
// Slope connects (x=7.5, z=0) and (x=0.25, z=15).
// H_apex = 15 * (7.5 / (7.5 - 0.25)) = 15.517241379... meters.

$fn = 4; // Set global resolution low for square generation optimization

// --- PARAMETERS ---
STONE_SIZE = 0.500; // Meters
NUM_LAYERS = 30;    // Calculated max height
BASE_WIDTH = NUM_LAYERS * STONE_SIZE;
CORE_HEIGHT = NUM_LAYERS * STONE_SIZE;

// Calculate geometric apex to ensure flush coverage of top stone and bottom corner
// Similar triangles: Apex_H / Base_Radius = Core_H / (Base_Radius - Top_Radius)
APEX_HEIGHT = CORE_HEIGHT * (BASE_WIDTH/2) / ((BASE_WIDTH/2) - (STONE_SIZE/2));

// --- MAIN RENDER ---

group() {
    // 1. The Stone Core
    // Rendered as slab layers for CAD optimization (instead of 9455 individual cubes)
    color([0.6, 0.6, 0.65]) 
        Core_Structure();

    // 2. The Marble Facade
    // Calculated as the boolean difference between the ideal smooth pyramid and the stepped core
    color([0.95, 0.95, 0.9, 0.6]) // Slightly transparent to show core in preview
        Marble_Facade();
}

// --- MODULES ---

module Core_Structure() {
    // Generates the stepped pyramid core
    // i is the layer number from the TOP (1 is the single top stone)
    for (i = [1 : NUM_LAYERS]) {
        layer_width = i * STONE_SIZE;
        // Z position: 
        // Total height is CORE_HEIGHT.
        // Layer 1 (top) sits from (CORE_HEIGHT - STONE_SIZE) to CORE_HEIGHT
        // Layer i sits at CORE_HEIGHT - (i * STONE_SIZE)
        z_pos = CORE_HEIGHT - (i * STONE_SIZE);
        
        translate([0, 0, z_pos]) {
            // Center the layer so corners align diagonally
            translate([0, 0, STONE_SIZE/2])
                cube([layer_width, layer_width, STONE_SIZE], center=true);
        }
    }
}

module Marble_Facade() {
    difference() {
        // The ideal smooth pyramid shape
        // Linear extrude a square to a point (scale=0)
        linear_extrude(height = APEX_HEIGHT, scale = 0) {
            square(BASE_WIDTH, center=true);
        }
        
        // Subtract the core to leave only the wedges
        // Expand core slightly (epsilon) to prevent z-fighting in preview
        epsilon = 0.001;
        translate([0,0,-epsilon]) 
            scale([1, 1, 1.001]) // Scale Z slightly to ensure clean boolean cut
            Core_Structure();
    }
}

// --- VIEW HELPER ---
// Add axes to confirm alignment (+Y is North)
module Axes() {
    color("red") arrow([2, 0.05, 0.05]);   // X
    color("green") arrow([0.05, 2, 0.05]); // Y (North)
    color("blue") arrow([0.05, 0.05, 2]);  // Z
}
```
