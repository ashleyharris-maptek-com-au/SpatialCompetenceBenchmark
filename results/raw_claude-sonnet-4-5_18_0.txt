```openscad
// Pyramid for Pharaoh - 20,000 stones, 50cm cubes
// North is +Y axis, centered at origin

stone_size = 500; // 50cm in mm
total_stones = 20000;

// Calculate maximum pyramid layers
// Layer n has n^2 stones, sum from 1 to h: h(h+1)(2h+1)/6
// Solving h(h+1)(2h+1)/6 = 20000 gives h ≈ 31.65
// So maximum full layers = 31, using 31*32*63/6 = 10,416 stones

layers = 31;

module pyramid_core() {
    for (layer = [0:layers-1]) {
        level_stones = layers - layer;
        z_pos = layer * stone_size;
        
        // Center offset for this layer
        offset = (level_stones - 1) * stone_size / 2;
        
        for (x = [0:level_stones-1]) {
            for (y = [0:level_stones-1]) {
                translate([
                    x * stone_size - offset,
                    y * stone_size - offset,
                    z_pos
                ])
                cube([stone_size, stone_size, stone_size]);
            }
        }
    }
}

module pyramid_facade() {
    base_size = layers * stone_size;
    height = layers * stone_size;
    
    // Create smooth pyramid hull
    hull() {
        // Base corners
        translate([-base_size/2, -base_size/2, 0])
            cylinder(r=0.1, h=0.1, $fn=4);
        translate([base_size/2, -base_size/2, 0])
            cylinder(r=0.1, h=0.1, $fn=4);
        translate([-base_size/2, base_size/2, 0])
            cylinder(r=0.1, h=0.1, $fn=4);
        translate([base_size/2, base_size/2, 0])
            cylinder(r=0.1, h=0.1, $fn=4);
        
        // Apex
        translate([0, 0, height])
            cylinder(r=0.1, h=0.1, $fn=4);
    }
}

module marble_facade() {
    base_size = layers * stone_size;
    height = layers * stone_size;
    
    difference() {
        pyramid_facade();
        
        // Remove internal core volume plus tolerance
        translate([0, 0, -1])
            pyramid_core();
    }
}

// Render marble facade only (comment/uncomment as needed)
marble_facade();
```
