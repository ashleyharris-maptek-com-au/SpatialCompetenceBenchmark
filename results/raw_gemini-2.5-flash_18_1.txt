```openscad
// OpenSCAD Plan for the Pharoh's Pyramid

// This plan details the construction of the tallest possible pyramid given the pharaoh's
// budget of 150,000 large stones, each a 50cm side-length cube.
// The pyramid is centered over the tomb, with its faces aligned to the cardinal directions
// (North being +Y, South -Y, East +X, West -X).

// --- Project Constants ---
// The side length of a single stone cube in meters.
STONE_SIZE = 0.5; // 50 centimeters

// The maximum number of cut large stones available for the project.
MAX_STONES_BUDGET = 150000;

// --- Calculate Pyramid Dimensions ---
// The specific stacking rule: "When placed on top of other stones, each stone must have
// all 4 base corners sitting in the middle of another stone."
// This implies a stepped pyramid structure where each successive layer (moving upwards)
// has one fewer stone along its side length compared to the layer directly below it.
//
// If the base layer (layer k=0) has `N` stones along one side, then:
// - Layer 0 (base): `N x N` stones
// - Layer 1: `(N-1) x (N-1)` stones
// - ...
// - Layer (N-1) (apex): `1 x 1` stone
//
// The total number of stones in such a pyramid is the sum of squares from 1 to N:
// Sum(i^2 for i=1 to N) = N * (N + 1) * (2 * N + 1) / 6.
//
// We need to find the largest integer `N` (which we'll call `BASE_N_STONES`)
// such that the total number of stones does not exceed `MAX_STONES_BUDGET`.
//
// Let's test values for N:
// - For N = 76:
//   Total stones = 76 * (76 + 1) * (2 * 76 + 1) / 6
//                = 76 * 77 * 153 / 6
//                = 149,896 stones. This is within the budget.
// - For N = 77:
//   Total stones = 77 * (77 + 1) * (2 * 77 + 1) / 6
//                = 77 * 78 * 155 / 6
//                = 155,055 stones. This exceeds the budget.
//
// Therefore, the maximum number of stones along one side of the base layer is 76.

BASE_N_STONES = 76; // Number of stones along one side of the base layer.
TOTAL_STONES_USED = BASE_N_STONES * (BASE_N_STONES + 1) * (2 * BASE_N_STONES + 1) / 6; // Confirmed: 149896

// Overall dimensions of the pyramid based on the calculated BASE_N_STONES and STONE_SIZE.
// The base width and height are directly proportional to the number of layers/stones.
PYRAMID_BASE_WIDTH = BASE_N_STONES * STONE_SIZE; // Total width/depth of the base in meters (76 * 0.5m = 38m).
PYRAMID_HEIGHT = BASE_N_STONES * STONE_SIZE;     // Total height of the pyramid in meters (76 * 0.5m = 38m).

// Offset to center the pyramid base at (0,0) in the XY-plane.
// The base layer will extend from -PYRAMID_BASE_WIDTH/2 to +PYRAMID_BASE_WIDTH/2
// along both the X and Y axes.
BASE_OFFSET_XY = -PYRAMID_BASE_WIDTH / 2; // -38m / 2 = -19m

// --- Pyramid Construction ---
// This section generates the individual stone cubes that form the pyramid.
// The plan ensures every step is visible, providing precise dimensions for the stonemason
// to calculate the marble facade wedges.

// Iterate through each layer of the pyramid.
// `k` represents the layer index, starting from 0 for the very base layer
// up to `BASE_N_STONES - 1` for the single stone at the apex.
for (k = [0 : BASE_N_STONES - 1]) {
    // Determine the number of stones along one side for the current layer.
    // This value decreases by one for each higher layer.
    current_layer_n_stones = BASE_N_STONES - k;

    // Calculate the X and Y translation offset for the current layer's grid.
    // According to the stacking rule, each higher layer is shifted by `STONE_SIZE / 2`
    // relative to the layer below it, towards the center.
    // This `layer_offset_xy` effectively defines the bottom-left corner
    // of the grid of stones for the current layer.
    layer_offset_xy = BASE_OFFSET_XY + k * STONE_SIZE / 2;

    // Calculate the Z-height for the current layer.
    // Each layer adds exactly one `STONE_SIZE` to the total height.
    layer_z = k * STONE_SIZE;

    // Generate individual stone cubes for the current layer.
    // `i` and `j` iterate through the grid positions within the current layer.
    for (i = [0 : current_layer_n_stones - 1]) {
        for (j = [0 : current_layer_n_stones - 1]) {
            // Calculate the bottom-left corner (x,y,z) for the current stone cube.
            stone_x = layer_offset_xy + i * STONE_SIZE;
            stone_y = layer_offset_xy + j * STONE_SIZE;
            
            // Translate and render a single stone cube at its calculated position.
            // All coordinates are in meters (SI units).
            translate([stone_x, stone_y, layer_z]) {
                cube(STONE_SIZE);
            }
        }
    }
}
```
