// Tallest pyramid with <= 600,000 stones (each stone is a 500 mm cube)
// Coordinate system: x east-west, y north-south (+y is north), z up
// Pyramid is centered over the tomb at (0,0,0), base lies on z = 0

n = 121;              // Stones per side at the base, and number of levels
L = 500;              // mm, edge length of one stone (50 cm)
stones = n*(n+1)*(2*n+1)/6;  // Total stones used = 597861 (must be <= 600000)

module pyramid_steps(n, L) {
    union() {
        // Level k (0-based): square slab of (n-k) stones per side
        // Each level is offset by L/2 in +x and +y, and by L in +z
        for (k = [0 : n-1]) {
            translate([ (k*L)/2, (k*L)/2, k*L ])
                cube([ (n-k)*L, (n-k)*L, L ], center = false);
        }
    }
}

// Center the pyramid over the tomb at (0,0,0)
translate([ -n*L/2, -n*L/2, 0 ])
    pyramid_steps(n, L);
