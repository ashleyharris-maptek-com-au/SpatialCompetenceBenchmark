// OpenSCAD Plan for Pharaoh's Pyramid
// Units: Metric SI (Meters)
//
// --- ENGINEERING CALCULATIONS ---
//
// 1. RESOURCE CONSTRAINTS:
//    - Stone Budget: 600,000 stones.
//    - Stone Size: 0.5m x 0.5m x 0.5m cube.
//    - Structural Requirement: Layer N (from top) requires N x N grid of stones.
//
// 2. MAX HEIGHT CALCULATION:
//    - Total stones for H layers = Sum of i^2 for i=1 to H.
//    - Formula: S = H(H+1)(2H+1)/6
//    - Solving for H where S <= 600,000:
//      - H=120: 583,220 stones.
//      - H=121: 597,901 stones. (OPTIMAL)
//      - H=122: 612,745 stones (Exceeds budget).
//    - Max Layers: 121
//    - Stones Used: 597,901
//    - Stones Remaining: 2,099
//
// 3. GEOMETRY:
//    - Base Layer (121): 121x121 stones. Width = 121 * 0.5m = 60.5m.
//    - Top Layer (1): 1x1 stone. Width = 0.5m.
//    - Step Height: 0.5m per layer.
//    - Step Indentation: 0.25m per layer (Slope 2:1).
//
// 4. MARBLE FACADE PLAN:
//    - The marble must cover the external stones to form a smooth pyramid.
//    - To cover the 'steps' created by the cubic stones, the smooth plane must 
//      pass through the top-outer edge of every stone layer.
//    - Bottom-most stone edge (Layer 121): x = +/- 30.25m, z = 0.5m.
//    - Top-most stone edge (Layer 1): x = +/- 0.25m, z = 60.5m.
//    - Slope Calculation: dz/dx = (60.5 - 0.5) / (0.25 - 30.25) = 60 / -30 = -2.
//    - Projected Base (at z=0): x = 30.25 + (0.5 / 2) = 30.5m. Total Base Width = 61.0m.
//    - Projected Apex (at x=0): z = 0.5 + (30.25 * 2) = 61.0m.
//
// 5. VOLUMES:
//    - Stone Volume: 597,901 * 0.125 m^3 = 74,737.625 m^3
//    - Smooth Pyramid Volume: (1/3) * Base_Area * Height = (1/3) * (61^2) * 61 = 75,653.667 m^3
//    - Marble Volume Required: 75,653.667 - 74,737.625 = 916.042 m^3

// --- PARAMETERS ---

stone_side = 0.5; // Meters
layers = 121;     // Calculated max height

// Facade Geometry
smooth_base_width = 61.0; 
smooth_height = 61.0;

// Visualization settings
$fn = 4; // Low polygon count for pyramids/cubes is sufficient and fast

// --- MODULES ---

module stone_core() {
    color([0.8, 0.7, 0.5, 1.0]) // Sandstone color
    union() {
        // We generate slabs per layer instead of individual cubes
        // to keep the rendering optimized for the stonemason's software.
        // N layers, i goes from 1 (top) to 121 (bottom)
        for (i = [1 : layers]) {
            translate([0, 0, (layers - i) * stone_side])
            linear_extrude(height = stone_side)
            square(size = i * stone_side, center = true);
        }
    }
}

module marble_facade() {
    color([0.95, 0.95, 1.0, 0.4]) // Marble white, semi-transparent to see structure
    difference() {
        // The smooth outer pyramid
        linear_extrude(height = smooth_height, scale = 0)
        square(size = smooth_base_width, center = true);
        
        // Subtract the stone core to create the wedges
        stone_core();
    }
}

module tomb_marker() {
    // Marker for the center of the tomb (Origin)
    color("red")
    translate([0,0,-1])
    cube([1,1,1], center=true);
}

// --- ASSEMBLY ---

// Ensure North is +Y (Standard OpenSCAD Y axis)
// The square base is aligned with X/Y axes.

group() {
    // 1. The Structure
    stone_core();
    
    // 2. The Facade (Wedges)
    marble_facade();
    
    // 3. Reference to Origin
    // tomb_marker(); // Optional, hidden for final plan
}

echo("Base Width (Marble):", smooth_base_width, "m");
echo("Total Height:", smooth_height, "m");
echo("Stone Volume:", 74737.625, "m^3");
echo("Marble Volume:", 916.042, "m^3");
