// Tallest pyramid core and marble casing plan
// Units: millimetres (SI, derived)
// Coordinate system: 
//   +z = up, 
//   +y = north (one face is aligned to +y), 
//   origin [0,0,0] at the centre of the base (tomb directly below).

// ------------------------------------------------------------
// Global parameters
// ------------------------------------------------------------

stone_size   = 500;    // mm, edge length of one structural stone (0.5 m)
layers       = 195;    // number of stepped layers in the stone core

core_base    = layers * stone_size;        // 97,500 mm
core_height  = layers * stone_size;        // 97,500 mm

// Outer smooth marble pyramid chosen to fully enclose the stepped core.
// Minimal symmetric right square pyramid that contains the entire core:
//   base = (layers + 1) * stone_size = 98,000 mm
//   height = (layers + 1) * stone_size = 98,000 mm
outer_base   = (layers + 1) * stone_size;
outer_height = (layers + 1) * stone_size;

// Derived values (for reference / volume calculations inside OpenSCAD):
stone_count  = layers * (layers + 1) * (2 * layers + 1) / 6;  // sum of squares
core_volume  = stone_count * stone_size * stone_size * stone_size;
// core_volume in m^3 would be: core_volume / 1e9

// ------------------------------------------------------------
// Stone core: stepped pyramid built from large stones
// Optimised representation: one solid slab per layer
// (each slab equals the union of all stones in that layer).
// ------------------------------------------------------------

module stone_core()
{
    // From bottom (largest square) to top (single-square cap).
    // Layer index "level" runs 0 .. layers-1.
    union() {
        for (level = [0 : layers - 1]) {
            layer_width = (layers - level) * stone_size;
            translate(
                [ -layer_width / 2,
                  -layer_width / 2,
                  level * stone_size ]
            )
                cube(
                    [ layer_width, layer_width, stone_size ],
                    center = false
                );
        }
    }
}

// ------------------------------------------------------------
// Smooth outer marble pyramid
// Right square pyramid, centred over the tomb, facing north (+y).
// ------------------------------------------------------------

module smooth_marble_pyramid()
{
    base   = outer_base;
    height = outer_height;

    // Points indexed for clarity:
    // 0: south-west, 1: south-east, 2: north-east, 3: north-west, 4: apex.
    polyhedron(
        points = [
            [ -base/2, -base/2, 0 ],  // 0: SW corner (south, west)
            [  base/2, -base/2, 0 ],  // 1: SE corner (south, east)
            [  base/2,  base/2, 0 ],  // 2: NE corner (north, east; +y face)
            [ -base/2,  base/2, 0 ],  // 3: NW corner (north, west)
            [ 0,        0,      height ] // 4: apex, directly above tomb
        ],
        faces = [
            [0, 1, 4],  // south face
            [1, 2, 4],  // east  face
            [2, 3, 4],  // north face (+y)
            [3, 0, 4],  // west  face
            [3, 2, 1, 0] // base (bottom square)
        ]
    );
}

// ------------------------------------------------------------
// Marble casing volume
// (the wedges of marble between the stepped stone core 
//  and the smooth outer pyramid).
// This is provided as a module; compute or export as needed.
// ------------------------------------------------------------

module marble_casing()
{
    difference() {
        smooth_marble_pyramid();
        stone_core();
    }
}
