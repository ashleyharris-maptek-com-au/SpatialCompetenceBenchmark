module sand_pile() {
    // Convert initial sand volume to cubic meters
    initial_sand = 5 * 0.764555; // 5 cubic yards to cubic meters
    // Convert piped sand volume to cubic meters
    piped_sand = 1500 * 0.0283168; // 1500 cubic feet to cubic meters
    // Convert train sand volume to cubic meters
    train_sand = 100;
    // Convert drum sand volume to cubic meters
    drum_sand = 20 * 44 * 0.00454609; // 20 drums of 44 UK gallons to cubic meters
    
    // Total sand volume in cubic meters
    total_sand = initial_sand + piped_sand + train_sand + drum_sand;
    
    // Silo diameter in meters
    silo_diameter = 10 * 0.9144; // 10 yards to meters
    silo_radius = silo_diameter / 2;
    
    // Height of sand fall in meters
    fall_height = 200 * 0.3048; // 200 feet to meters
    
    // Angle of repose in radians
    angle_of_repose = 33 * (PI / 180);
    
    // Calculate the cone height based on volume and angle of repose
    cone_radius = silo_radius;
    cone_height = (3 * total_sand) / (PI * pow(cone_radius, 2));
    
    // Ensure the cone height is within the silo constraints
    max_height = fall_height + cone_height;
    
    // Create the cone shape
    translate([0, 0, fall_height])
    cylinder(h = max_height, r1 = 0, r2 = cone_radius, $fn = 100);
}

sand_pile();
