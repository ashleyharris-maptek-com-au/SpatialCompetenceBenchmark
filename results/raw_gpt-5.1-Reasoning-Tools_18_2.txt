/*
 Tallest solid stepped pyramid the pharaoh can build
 given 600,000 cubic stones of 0.5 m edge length.

 Coordinate system:
   - x axis: east  (+x)
   - y axis: north (+y)
   - z axis: up    (+z)
 The square base is centered at the origin (0,0,0).
*/

stone_side_mm = 500;      // Edge length of one stone cube, in millimetres.
levels        = 121;      // Stones along one edge of the lowest level.
max_stones    = 600000;   // Available stones.
total_stones  = levels*(levels+1)*(2*levels+1)/6;  // 597861 stones used.

base_side_mm  = levels * stone_side_mm;
height_mm     = levels * stone_side_mm;

// Core stepped pyramid built from solid square "slabs",
// one slab per stone layer. This has exactly the same volume
// and external stepped profile as stacking individual stones
// with no internal voids.
module stone_core() {
    union() {
        for (level = [0 : levels-1]) {
            stones_per_side = levels - level;                 // Stones along one side at this level.
            side_mm = stones_per_side * stone_side_mm;        // Physical side length at this level.
            translate([-side_mm/2, -side_mm/2, level*stone_side_mm])
                cube([side_mm, side_mm, stone_side_mm], center=false);
        }
    }
}

// Optional: idealised smooth pyramid sharing the same
// base and apex as the stepped stone core.
// Useful as a geometric reference for marble facade design.
module smooth_pyramid() {
    polyhedron(
        points = [
            [ 0,              0,              height_mm],   // 0: apex
            [-base_side_mm/2, -base_side_mm/2, 0],          // 1: south-west  corner
            [ base_side_mm/2, -base_side_mm/2, 0],          // 2: south-east  corner
            [ base_side_mm/2,  base_side_mm/2, 0],          // 3: north-east  corner
            [-base_side_mm/2,  base_side_mm/2, 0]           // 4: north-west  corner
        ],
        faces = [
            [1,2,3,4],   // base
            [0,1,2],     // south face  (towards -y)
            [0,2,3],     // east  face  (towards +x)
            [0,3,4],     // north face  (towards +y)
            [0,4,1]      // west  face  (towards -x)
        ]
    );
}

render_mode = 0;

if (render_mode == 0)
    stone_core();

if (render_mode == 1)
    smooth_pyramid();

if (render_mode == 2)
    union() {
        color([0.8,0.7,0.5]) stone_core();
        color([1,1,1])       smooth_pyramid();
    }
