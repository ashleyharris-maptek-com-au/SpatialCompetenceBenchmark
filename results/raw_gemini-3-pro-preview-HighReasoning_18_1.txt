```openscad
// OpenSCAD Plan for Pharaoh's Tomb Pyramid
// ---------------------------------------------------------
// UNIT: Meters (SI)
//
// DESIGN CONSTRAINTS & CALCULATIONS:
// 1. Stone Dimensions: 0.5m x 0.5m x 0.5m Cube (Volume = 0.125 m^3)
// 2. Budget: 150,000 Stones.
// 3. Structural Constraint: Every stone must rest with 4 corners on centers of stones below.
//    This dictates a standard arithmetic progression of square layers:
//    Top layer (1): 1x1 stones.
//    Layer (2): 2x2 stones.
//    Layer (n): n*n stones.
//
// 4. Max Height Calculation:
//    Total Stones S = Sum(n^2) for n=1 to N = n(n+1)(2n+1)/6
//    Solving for S <= 150,000:
//    N = 76 Layers.
//    Stones Used = 149,226.
//    Stones Remaining = 774 (Insufficient for layer 77 which requires 77^2 = 5,929 stones).
//
// 5. Manpower Check:
//    100,000 Men / 10 Men per stone = 10,000 simultaneous stone capacity.
//    Since construction is sequential, the total stone budget (150k) is the limiting factor
//    for the final volume, not the instantaneous carrying capacity.
//
// 6. Facade:
//    Marble wedges fill the steps to create a smooth square pyramid.
//    Slope geometry: 2:1 (Rise 0.5m, Run 0.25m per step).
// ---------------------------------------------------------

// --- Global Parameters ---
stone_side = 0.500; // Meters
layers = 76;
stones_used = 149226;

// Dimensions derived from layers
base_width_stones = layers;
base_width_m = base_width_stones * stone_side;
total_height_m = layers * stone_side;

// --- Viewport Setup ---
$fn = 4; // Low poly count for precise square pyramids
$vpr = [60, 0, 45]; // Isometric-like view
$vpt = [0, 0, total_height_m/2];
$vpd = total_height_m * 4;

// --- Information Output (Console) ---
echo(str("--------------------------------------------------"));
echo(str("PHARAOH'S PYRAMID CONSTRUCTION PLAN"));
echo(str("--------------------------------------------------"));
echo(str("Max Layers: ", layers));
echo(str("Total Height: ", total_height_m, "m"));
echo(str("Base Width: ", base_width_m, "m"));
echo(str("Stones Used: ", stones_used));
echo(str("Stones Remaining: ", 150000 - stones_used));
echo(str("Core Volume: ", stones_used * pow(stone_side, 3), "m^3"));
echo(str("--------------------------------------------------"));

// --- Modules ---

module stone_core() {
    // Generates the stepped sandstone core.
    // Optimized: Renders each layer as a single slab rather than thousands of cubes.
    // This maintains precise volume and dimensions while allowing CAD software to function.
    
    color([0.85, 0.75, 0.6]) // Sandstone color
    for (i = [0 : layers - 1]) {
        // i = 0 is the bottom layer (widest)
        // stones_in_width = layers - i
        
        layer_width = (layers - i) * stone_side;
        
        // Z position: i * stone_side
        translate([0, 0, (i * stone_side) + (stone_side / 2)])
            cube([layer_width, layer_width, stone_side], center = true);
    }
}

module marble_facade() {
    // Generates the smooth marble casing.
    // The facade touches the outer corners of every stone step.
    // It is shown transparently to reveal the core structure.
    
    color([0.95, 0.95, 1.0, 0.4]) // White Marble, Transparent
    
    // The smooth pyramid base is exactly the width of the bottom layer.
    // The apex meets at a point exactly one stone_height above the center of the top stone's base.
    // Slope = stone_side / (stone_side/2) = 2.
    // Height = Base_Width * Slope / 2 = (N*s) * 1 = N*s.
    
    linear_extrude(height = total_height_m, scale = 0)
        square(base_width_m, center = true);
}

module project_info() {
    // 3D Text annotation for the Stonemason
    color("black")
    translate([base_width_m/2 + 2, -base_width_m/2, 0]) {
        linear_extrude(0.1) {
            text("PROJECT: ETERNAL REST", size=2);
            translate([0, -3]) text(str("Height: ", total_height_m, "m"), size=1.5);
            translate([0, -5]) text(str("Base: ", base_width_m, "m"), size=1.5);
            translate([0, -7]) text(str("Layers: ", layers), size=1.5);
            translate([0, -9]) text("Orientation: +Y (North)", size=1.5);
        }
    }
    
    // Compass / Orientation Arrow
    translate([0, base_width_m/2 + 5, 0]) {
        color("red") {
            linear_extrude(0.5) {
                text("N", halign="center", valign="center", size=4);
            }
        }
        translate([0, -4, 0])
            color("red")
            cube([1, 6, 0.5], center=true);
        translate([0, -1, 0])
            color("red")
            cylinder(h=0.5, r1=2, r2=0, $fn=3, center=true);
    }
}

// --- Render Geometry ---

// 1. The structural core (Stones)
stone_core();

// 2. The marble cladding (Casing)
// To see the specific wedges required, perform a difference() 
// between marble_facade() and stone_core(). 
// Here we overlay them for the plan view.
marble_facade();

// 3. Plan Annotations
project_info();
```
