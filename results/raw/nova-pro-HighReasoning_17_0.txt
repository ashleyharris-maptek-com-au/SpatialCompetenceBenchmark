// Conversion factors
yd_to_m = 0.9144;
ft_to_m = 0.3048;
gal_to_m3 = 0.00454609;

// Initial volume in cubic meters
initial_volume_m3 = 5 * yd_to_m^3;

// Added volume from trucks in cubic meters
truck_volume_m3 = 1500 * ft_to_m^3;

// Added volume from train in cubic meters
train_volume_m3 = 100;

// Added volume from drums in cubic meters
drum_volume_m3 = 20 * 44 * gal_to_m3;

// Total volume in cubic meters
total_volume_m3 = initial_volume_m3 + truck_volume_m3 + train_volume_m3 + drum_volume_m3;

// Silo diameter in meters
silo_diameter_m = 10 * yd_to_m;

// Angle of repose in radians
angle_of_repose_rad = 33 * (3.14159 / 180);

// Height of the cone-shaped pile
height = total_volume_m3 / (silo_diameter_m * tan(angle_of_repose_rad) * 3.14159 / 4);

// OpenSCAD shape definition
module sand_pile() {
  difference() {
    cylinder(h = height, r = silo_diameter_m / 2, $fn = 32);
    translate([0, 0, height]) 
      cylinder(h = 200 * ft_to_m, r = silo_diameter_m / 2, $fn = 32);
  }
}

sand_pile();
