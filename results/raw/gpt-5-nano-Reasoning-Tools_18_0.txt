// Pharaoh's Pyramid OpenSCAD plan
// All dimensions in millimeters (SI units)
//
// Assumptions handled by the model:
// - Each stone is a 50 mm cube.
// - Stacking rule: layer L (0-based) is offset by L*(s/2) in x and y, and sits at z = L*s.
// - Bottom layer contains HEIGHT x HEIGHT stones; each subsequent layer has one fewer stone per side.
// - Height (number of layers) is maximized given 20,000 stones total: sum_{k=1}^{HEIGHT} k^2 <= 20000.
// - The pyramid faces north (+Y) and is centered over the tomb by construction.
// - No internal voids other than the discrete blocks themselves.

N_STONES = 20000;          // budget stones
STONE = 50;                 // stone side length in mm (50 mm cube)

function sumSquares(n) = int((n*(n+1)*(2*n+1))/6); // sum of squares 1^2 + 2^2 + ... + n^2

// Binary search to find the largest HEIGHT such that sumSquares(HEIGHT) <= N_STONES
function height_bs(n, lo, hi) =
  (lo > hi) ? hi :
  let(mid = floor((lo + hi) / 2))
    (sumSquares(mid) <= n) ? height_bs(n, mid + 1, hi) : height_bs(n, lo, mid - 1);

HEIGHT = height_bs(N_STONES, 0, 100); // maximum base side length in stones per side

// Build the pyramid: HEIGHT layers; base has HEIGHT x HEIGHT stones
module pyramid_build() {
  s = STONE;
  // Layers from bottom (L=0) to top (L=HEIGHT-1)
  for (L = [0 : 1 : HEIGHT - 1]) {
    N = HEIGHT - L; // number of stones per side on this layer
    // Create N x N stones for this layer
    for (i = [0 : 1 : N - 1]) {
      for (j = [0 : 1 : N - 1]) {
        translate([ i*s + L*(s/2), j*s + L*(s/2), L*s ])
          cube([s, s, s], center = false);
      }
    }
  }
}

pyramid_build();
