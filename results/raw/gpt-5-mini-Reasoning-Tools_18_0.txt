// Pyramid plan (OpenSCAD)
// Units: meters (SI)
// ------------------------------------------------------------
// Pharaoh's resources:
//  - Available large cut stones: 20,000
//  - Large stone: 0.50 m side (cube)
//  - Each stone volume = 0.5^3 = 0.125 m^3
//  - Constraint for stacking: each stone in an upper layer sits with its 4 base corners
//    over the centers of four stones below; layers therefore have sizes M x M
//    with M decreasing by 1 each layer (a stepped square pyramid).
//
// Result (optimal tallest pyramid within 20,000 stones):
//  - Number of layers (base side count) N = 38
//  - Total stones required = sum_{k=1..38} k^2 = 19,096 stones (<= 20,000 available)
//  - Pyramid height = 38 * 0.5 m = 19.000 m
//  - Total solid volume = 19,096 * 0.125 m^3 = 2,387.000 m^3
//  - Orientation: north is +Y; pyramid is centered at (0,0) so tomb is at origin.
// ------------------------------------------------------------
// Model strategy (performance + precision):
//  - Each layer consists of M x M stones (M = N - L for layer L=0..N-1).
//  - Internals of each layer (non-exposed stones) are merged into one single
//    rectangular block per layer to reduce primitives while preserving exact
//    volume. Only the perimeter stones (the externally visible stones) are
//    modelled individually as full cubes so the stonemason can see every
//    external stone to plan marble wedges.
//  - This representation is exact: union of the interior blocks and the
//    perimeter cubes equals the union of all original cubes (no overlaps),
//    so the solid volume matches stones*stone_volume exactly (accurate to mm).
// ------------------------------------------------------------

// Parameters (edit here if different stone size or stone budget required)
stone_side = 0.5;          // meters (500 mm)
N = 38;                    // layers (base is N x N stones)
plate_thickness = 0.001;   // not used for final geometry; kept if needed
// A tiny epsilon to avoid accidental boolean-touch artifacts in some viewers
eps = 1e-12;

// Derived values (kept as constants so numbers are explicit)
stones_used = N*(N+1)*(2*N+1)/6;                      // sum k^2
stone_volume = stone_side * stone_side * stone_side;  // m^3 per stone
total_volume = stones_used * stone_volume;            // m^3 total
pyramid_height = N * stone_side;                      // meters
base_width = N * stone_side;                          // meters

// Echoing key values to the OpenSCAD console (when the file is run in OpenSCAD)
echo(str("stone_side (m): ", stone_side));
echo(str("layers N: ", N));
echo(str("stones used: ", stones_used));
echo(str("stone volume (m^3): ", stone_volume));
echo(str("total pyramid volume (m^3): ", total_volume));
echo(str("pyramid height (m): ", pyramid_height));
echo(str("base width (m): ", base_width));

// Visual color scheme (purely visual helpers)
color_interior = [0.88, 0.82, 0.75];   // interior block color (sandstone look)
color_perim   = [0.70, 0.60, 0.50];   // perimeter stones color
color_north    = [0.95, 0.55, 0.45];  // highlight for north-facing perimeter stones
color_tomb     = [1,0,0];

module perimeter_cube(x, y, zc, c = color_perim) {
    color(c)
        translate([x, y, zc])
            cube([stone_side, stone_side, stone_side], center=true);
}

module interior_block(M, zc) {
    // M >= 3 assumed
    inner_count = M - 2;
    size_xy = inner_count * stone_side;
    // If inner_count==0 this module should not be called.
    color(color_interior)
        translate([0, 0, zc])
            cube([size_xy, size_xy, stone_side], center=true);
}

module build_pyramid() {
    // Pyramid centered at origin, base layer L=0 (largest, M=N)
    for (L = [0 : N - 1]) {
        M = N - L;                      // stones per side on this layer
        z_center = (L + 0.5) * stone_side; // center height of cubes in this layer

        // Interior block (merge inner (M-2)x(M-2) stones into a single block)
        if (M >= 3) {
            interior_block(M, z_center);
        }

        // Place perimeter stones individually so every external stone is visible
        // Loop indices i, j correspond to stones indices 0..M-1 along X and Y
        for (i = [0 : M - 1]) {
            // compute X coordinate of this stone's center, accurate to mm
            xi = (i - (M - 1) / 2) * stone_side;
            for (j = [0 : M - 1]) {
                // only perimeter stones (exposed externally) are instantiated here
                if ((i == 0) || (i == M - 1) || (j == 0) || (j == M - 1)) {
                    yj = (j - (M - 1) / 2) * stone_side;
                    // Highlight stones on the north-facing outer row (j == M-1)
                    if (j == M - 1) {
                        perimeter_cube(xi, yj, z_center, color_north);
                    } else {
                        perimeter_cube(xi, yj, z_center, color_perim);
                    }
                }
            }
        }
    }
}

// Small marker for tomb center (at origin), so CAD user sees alignment
module tomb_marker() {
    translate([0, 0, 0.01])
        color(color_tomb)
            cylinder(h=0.02, r=0.05, center=true);
}

// Coordinate helper: draw a small north arrow (points in +Y direction)
module north_arrow() {
    arrow_base_y = base_width/2 + 0.6; // place just north of base
    // vertical offset so arrow is visible above ground
    translate([0, arrow_base_y, 0.15]) {
        color([0,0.4,0.8])
            // shaft along +Y - use cylinder and cone oriented along Z then rotate
            // easier: build along Z then rotate -90 deg around X to point along +Y
            rotate([90,0,0]) {
                // shaft
                translate([0,0,-0.02]) cylinder(h=0.6, r=0.02, center=false);
                // head
                translate([0,0,0.6]) cone = cylinder(h=0.14, r1=0.06, r2=0, center=false);
            }
    }
}

// Render the pyramid and helpers
build_pyramid();
tomb_marker();
north_arrow();
