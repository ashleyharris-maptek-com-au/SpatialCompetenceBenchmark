// Pharaoh's Pyramid Plan
// Optimized for construction and visualization
// Units: Metric (Meters)

/* 
   ARCHITECTURAL SPECIFICATIONS:
   
   Resource Constraints:
   - Total Stone Budget: 2,500,000 blocks
   - Stone Dimensions: 0.5m x 0.5m x 0.5m cube
   - Structural Constraint: Canonical 4-corner support (Sum of Squares series)
   
   Calculations:
   - Max Height H satisfying Sum(n=1 to H, n^2) <= 2,500,000
   - H = 195 layers
   - Total Stones Used: 2,490,670
   - Remaining Stones: 9,330
   
   Dimensions:
   - Core Height: 97.5m
   - Core Base: 97.5m x 97.5m
   - Facade Base: 98.0m x 98.0m (Slope = 2.0)
   - Orientation: Faces North (+Y)
*/

// --- Parameters ---
stone_size = 0.500; // Meters
layers = 195;

// Calculated Dimensions
core_base_width = layers * stone_size;
core_height = layers * stone_size;

// Facade Geometry
// The smooth facade connects the outer base corner to the theoretical apex.
// To cover the steps where each step retreats stone_size/2 for every stone_size rise:
// The facade base must extend stone_size/2 outward from the bottom block.
facade_base_width = core_base_width + stone_size; 
facade_height = facade_base_width; // Geometric property of 2:1 slope steps

// --- Modules ---

module stone_core() {
    // Generates the stepped pyramid core
    // Optimized: Renders one large prism per layer instead of millions of cubes
    color([0.6, 0.6, 0.65]) // Granite Grey
    for (i = [0 : layers - 1]) {
        // Layer index 0 is bottom
        layer_width = (layers - i) * stone_size;
        elevation = i * stone_size;
        
        translate([0, 0, elevation + (stone_size / 2)])
            cube([layer_width, layer_width, stone_size], center = true);
    }
}

module marble_facade_volume() {
    // The theoretical smooth pyramid shape
    color([0.95, 0.95, 0.9, 0.4]) // Marble White, Translucent for plans
    linear_extrude(height = facade_height, scale = 0)
        square(facade_base_width, center = true);
}

module construction_plan() {
    // Render the Stone Core
    stone_core();
    
    // Render the Marble Facade Wedges
    // Calculated as the boolean difference between smooth shell and stepped core
    // This allows the stonemason to see exactly the volume required for cutting
    difference() {
        marble_facade_volume();
        stone_core();
    }
}

// --- Output ---

// Echo stats for the Stonemason
echo("--- PYRAMID PLAN STATISTICS ---");
echo("Total Layers:", layers);
echo("Total Stones Required:", 2490670);
echo("Core Base Width (m):", core_base_width);
echo("Total Height (m):", core_height);
echo("Marble Facade Base Width (m):", facade_base_width);
echo("-------------------------------");

// Generate Model
construction_plan();
