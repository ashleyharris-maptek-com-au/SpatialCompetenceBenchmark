// OpenSCAD plan for the Pharoh's pyramid
// Author: AI Assistant
// Date: 2023-10-27

// --- Project Parameters ---

// Stone dimensions in meters (50cm side-length cube)
stone_side = 0.5; // meters

// Number of available cut large stones
total_stone_budget = 20000;

// --- Calculation of Pyramid Dimensions ---

// The stacking rule "each stone must have all 4 base corners sitting in the middle of another stone"
// implies an offset step pyramid where each layer is shifted by half a stone_side in X and Y
// relative to the layer below it, and contains (N-1)x(N-1) stones if the layer below had NxN stones.
// The total number of stones for a pyramid with a base of S0 x S0 stones is the sum of squares:
// Sum(k^2 for k=1 to S0) = S0 * (S0 + 1) * (2*S0 + 1) / 6

// We need to find the largest S0 such that Sum(k^2) <= total_stone_budget.
// Let's test some values:
// S0 = 30 => 9455 stones
// S0 = 35 => 14910 stones
// S0 = 38 => 19019 stones
// S0 = 39 => 20540 stones (Exceeds budget)

// Therefore, the maximum number of stones on the side of the base layer is 38.
max_base_stones_on_side = 38; // S0
stones_used = max_base_stones_on_side * (max_base_stones_on_side + 1) * (2 * max_base_stones_on_side + 1) / 6;

// Pyramid height in meters
pyramid_height = max_base_stones_on_side * stone_side;

// Base width/depth in meters
pyramid_base_dimension = max_base_stones_on_side * stone_side;

// Display calculated parameters for the stonemason's reference
echo(STR("--- Pyramid Specifications ---"));
echo(STR("Stone side length: ", stone_side, " m"));
echo(STR("Max stones on base side (S0): ", max_base_stones_on_side));
echo(STR("Total stones used: ", stones_used));
echo(STR("Pyramid height: ", pyramid_height, " m"));
echo(STR("Pyramid base width/depth: ", pyramid_base_dimension, " m"));
echo(STR("--- End Specifications ---"));


// --- Module for a single stone ---
// This module defines a single cube representing a large stone.
module stone() {
    cube(stone_side);
}

// --- Main Pyramid Construction ---
// The pyramid is built layer by layer, starting from the base (layer_k = 0).
// Each layer is centered over the future tomb (0,0,0) and its base corners
// are offset to sit in the middle of the stones below, as per the rules.
// OpenSCAD's default orientation is +Z up, +X right, +Y back.
// The problem states +Y is North, so the pyramid will extend in +Y/-Y.

for (layer_k = 0; layer_k < max_base_stones_on_side; layer_k = layer_k + 1) {
    // Number of stones on one side for the current layer
    num_stones_on_side_current_layer = max_base_stones_on_side - layer_k;
    
    // Calculate the overall offset for the current layer to ensure the entire pyramid
    // is centered at (0,0,0) and each successive layer is shifted by half a stone.
    // The base layer (layer_k=0) has its bottom-left corner at (-pyramid_base_dimension/2, -pyramid_base_dimension/2, 0).
    // Each subsequent layer (layer_k) is shifted by `layer_k * stone_side / 2` in both X and Y.
    x_y_offset_for_layer = -pyramid_base_dimension / 2 + layer_k * stone_side / 2;
    
    // Z-coordinate for the bottom of the current layer
    z_position_for_layer = layer_k * stone_side;
    
    // Iterate through each stone within the current layer (i for X, j for Y)
    for (i = 0; i < num_stones_on_side_current_layer; i = i + 1) {
        for (j = 0; j < num_stones_on_side_current_layer; j = j + 1) {
            // Translate the stone to its precise position
            translate([x_y_offset_for_layer + i * stone_side, 
                       x_y_offset_for_layer + j * stone_side, 
                       z_position_for_layer]) {
                stone(); // Place a single stone cube
            }
        }
    }
}
