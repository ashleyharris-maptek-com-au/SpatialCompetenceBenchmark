// SPECIFICATIONS:

// - Layer offset: 250 mm per edge per layer

half_base = base_width / 2;

polyhedron(
    points = [
        [-half_base, -half_base, 0],
        [half_base, -half_base, 0],
        [half_base, half_base, 0],
        [-half_base, half_base, 0],
        [0, 0, pyramid_height]
    ],
    faces = [
        [0, 1, 2, 3],
        [0, 4, 1],
        [1, 4, 2],
        [2, 4, 3],
        [3, 4, 0]
    ]
);
