// OpenSCAD Plan for the Tomb of the Pharaoh
// Units: Metric SI (Meters)
//
// DESIGN CALCULATIONS:
// --------------------
// Stone Constraints:
// - Stone Size (s): 0.5m x 0.5m x 0.5m (Cube)
// - Available Stones: 600,000
// - Stacking Rule: Corners of top stone must rest on centers of stones below.
//   This necessitates a Sum of Squares packing sequence (1, 4, 9, 16...).
//   Layer n (from top) contains n^2 stones.
//
// Height Calculation:
//   Sum(n^2) for n=1 to H <= 600,000
//   H(H+1)(2H+1)/6 <= 600,000
//   Maximum Height H = 121 Layers.
//   Total Stones Used = 597,861.
//   Remaining Stones = 2,139.
//
// Dimensions:
// - Core Base Width: 121 * 0.5m = 60.5m
// - Core Height: 121 * 0.5m = 60.5m
// - Smooth Facade Geometry:
//   To cover the "steps" completely with a smooth pyramid, the facade planes
//   must pass through the top-outer edge of the top stone (layer 1)
//   and the top-outer edge of the bottom stone (layer 121).
//   This results in a pyramid of:
//   - Height: (121 + 1) * 0.5m = 61.0m
//   - Base Width: (121 + 1) * 0.5m = 61.0m
//
// Coordinate System:
// - Center at (0,0,0) (Tomb location)
// - +Y is North.

// Global Parameters
s = 0.5;          // Stone side length in meters
H = 121;          // Calculated max layers
marble_alpha = 0.4; // Transparency for the facade to show internal structure

// Main Assembly
union() {
    StoneCore();
    MarbleFacade();
}

module StoneCore() {
    // Renders the step pyramid core
    // Optimization: Renders one slab per layer representing the n*n grid of stones
    // This is computationally efficient and geometrically accurate for the envelope.
    color([0.5, 0.5, 0.5]) // Grey Granite
    for (i = [1 : H]) {
        // i is layer number from top (1 is top, H is bottom)
        // Layer Width = i * s
        // Layer Z center:
        //   Base of stack is Z=0.
        //   Top of stack is H*s.
        //   Layer i (from top) is the i-th block down.
        //   Z_center = (H - i) * s + s/2
        
        translate([0, 0, (H - i) * s + s / 2])
            cube([i * s, i * s, s], center = true);
    }
}

module MarbleFacade() {
    // Renders the smooth marble cladding
    // Calculated as the boolean difference between the ideal Smooth Pyramid
    // and the stepped Stone Core.
    
    // Smooth Pyramid Dimensions
    // Slope logic: Rise = s, Run = s/2 (Stepping out half a block per block down)
    // Slope gradient = 2.
    // Apex Z = (H + 1) * s
    // Base Width = (H + 1) * s
    pyr_h = (H + 1) * s;
    pyr_w = (H + 1) * s;
    
    // Radius for cylinder to creating a square base of width W
    // r = (W/2) * sqrt(2) = W / sqrt(2)
    pyr_r = pyr_w / sqrt(2);

    color([0.95, 0.95, 1.0, marble_alpha]) // White Marble
    difference() {
        // The Outer Smooth Pyramid
        // Rotated 45 degrees to align edges with N/S/E/W (X/Y axes)
        // cylinder with $fn=4 creates a diamond shape by default
        rotate([0, 0, 45])
            cylinder(h = pyr_h, r1 = pyr_r, r2 = 0, $fn = 4);
        
        // Subtract the Stone Core to leave only the marble wedges
        // Using a slight epsilon scale to avoid z-fighting on shared surfaces
        // though logically they touch exactly.
        StoneCore();
    }
}
