rs = 4.572;
tan33 = tan(33 * pi / 180);
vTotal = 5 * 0.764555 + 1500 * 0.028316846592 + 100 + 20 * 44 * 0.00454609;
vConeMax = (pi / 3) * tan33 * rs * rs * rs;

if (vTotal <= vConeMax) {
  r = pow((3 * vTotal) / (pi * tan33), 1/3);
  cylinder(h = r * tan33, r1 = 0, r2 = r);
} else {
  hCone = rs * tan33;
  vTop = vTotal - vConeMax;
  hTop = vTop / (pi * rs * rs);
  union() {
    cylinder(h = hCone, r1 = 0, r2 = rs);
    translate([0,0,hCone]) cylinder(h = hTop, r = rs);
  }
}
