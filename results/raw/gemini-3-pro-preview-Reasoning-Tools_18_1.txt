// OpenSCAD Plan for Pharaoh's Pyramid
// Units: Meters (SI)
// Constraints:
// - Max Stones: 150,000
// - Stone Size: 0.5m x 0.5m x 0.5m
// - Labor: 100,000 men (Not a limiting factor for stone count, max layer < 10,000 stones)
// - Structure: Staggered center-aligned stacking (Square Pyramid)
// - Geometry: North facing (+y), Centered at origin, Smooth marble facade

// ---------------------------------------------
// CALCULATIONS
// ---------------------------------------------
// The number of stones in a square pyramid of N layers is given by the sum of squares:
// Stones = N(N+1)(2N+1)/6
//
// Solving for Stones <= 150,000:
// N=76 implies 149,226 stones.
// N=77 implies 155,155 stones (Over budget).
// Max Layers = 76.
//
// Core Height = 76 layers * 0.5m = 38.0m.
// Core Base Width = 76 stones * 0.5m = 38.0m.
//
// Facade Geometry:
// To cover the steps smoothly with marble wedges, a smooth pyramid is circumscribed 
// touching the top-outer corners of every stone step.
// Slope calculation (Step Rise=0.5, Step Run=0.25) -> Slope = 2.
// Extending this slope from the top stone (width 0.5m at z=38m) to the apex:
// The facade must rise 0.5m above the top stone to resolve to a point.
// Facade Height = 38.5m.
// Facade Base Width = 38.5m.

$fn = 60; // Resolution for rendering

// ---------------------------------------------
// PARAMETERS
// ---------------------------------------------
stone_size = 0.500; // Meters
num_layers = 76;

// Derived Dimensions
core_height = num_layers * stone_size;
facade_height = core_height + stone_size;
facade_base = (num_layers * stone_size) + stone_size;

// ---------------------------------------------
// MODULES
// ---------------------------------------------

// The structural core made of granite blocks
module StoneCore() {
    color([0.85, 0.75, 0.60]) // Sandstone/Granite Color
    for (i = [1 : num_layers]) {
        // i = 1 is the top layer (0.5m wide)
        // i = 76 is the bottom layer (38m wide)
        
        layer_width = i * stone_size;
        
        // Calculate Z position
        // Bottom layer (i=76) sits at Z=0
        // Top layer (i=1) sits at Z=37.5
        z_pos = (num_layers - i) * stone_size;
        
        translate([0, 0, z_pos + stone_size/2])
            cube([layer_width, layer_width, stone_size], center=true);
    }
}

// The theoretical smooth shape of the final pyramid
module SmoothShell() {
    // Linear extrude with scale 0 creates a pyramid
    translate([0, 0, 0])
        linear_extrude(height = facade_height, scale = 0)
            square(facade_base, center=true);
}

// The marble wedges required to smooth the facade
module MarbleCladding() {
    color([0.95, 0.95, 1.0, 0.6]) // Marble White, slightly transparent for CAD view
    difference() {
        SmoothShell();
        StoneCore();
    }
}

// ---------------------------------------------
// RENDER
// ---------------------------------------------

// 1. The structural core
StoneCore();

// 2. The marble cladding (Wedges)
// Rendered transparently to verify the core is fully covered (no voids visible from outside)
// and to allow the mason to see the wedge shapes relative to the stones.
MarbleCladding();

// ---------------------------------------------
// DATA OUTPUT (Console)
// ---------------------------------------------
echo(str("Plan Specifications:"));
echo(str("Max Height: ", facade_height, "m"));
echo(str("Base Width: ", facade_base, "m"));
echo(str("Stone Layers: ", num_layers));
echo(str("Stones Used: ", 149226));
echo(str("Marble Volume: ", (1/3)*facade_base*facade_base*facade_height - 149226*pow(stone_size,3), " m3"));
