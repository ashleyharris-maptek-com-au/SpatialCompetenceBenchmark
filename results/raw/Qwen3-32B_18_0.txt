// OpenSCAD Plan for the Tallest Pyramid Given Constraints

// Parameters
stone_side = 0.5;          // 50cm
stones_per_layer = 0;      // Will be calculated
total_stones = 20000;      // Total stones available
men_per_stone = 10;        // 10 men per stone
total_men = 100000;        // Total men available for moving stones
max_stones_by_men = floor(total_men / men_per_stone); // Max stones we can move

// The pyramid must be square-based and each stone rests on 4 others
// So, the number of stones per layer follows the pattern N^2, (N-2)^2, (N-4)^2, etc.
// We need to find the largest N such that the total number of stones does not exceed the available stones

// Function to compute total stones needed for a pyramid of side length N (in stones)
function pyramid_stones(N) = 
  let (layers = floor((N + 1) / 2)) 
  sum([for (i = [0 : layers - 1]) (N - 2 * i)^2 ]);

// Binary search to find the maximum N pyramid we can build
function find_max_pyramid(max_stones) = 
  let (low = 1, high = max_stones, best = 1)
  while (low <= high) {
    mid = floor((low + high) / 2);
    stones_needed = pyramid_stones(mid);
    if (stones_needed <= max_stones) {
      best = mid;
      low = mid + 1;
    } else {
      high = mid - 1;
    }
  } best;

// Determine maximum pyramid side length in stone units
max_pyramid_side = find_max_pyramid(min(total_stones, max_stones_by_men));

// Calculate the actual pyramid layers and stones
layers = floor((max_pyramid_side + 1) / 2);
echo("Maximum Pyramid Side Length (in stones):", max_pyramid_side);
echo("Number of Layers:", layers);
echo("Total Stones Used:", pyramid_stones(max_pyramid_side));

// Function to generate the pyramid in OpenSCAD
function pyramid_module() = 
  for (layer = [0 : layers - 1]) {
    let (side_length = max_pyramid_side - 2 * layer)
    let (x_offset = - (side_length * stone_side) / 2)
    let (y_offset = - (side_length * stone_side) / 2)
    translate([x_offset, y_offset, layer * stone_side])
    for (x = [0 : side_length - 1]) {
      for (y = [0 : side_length - 1]) {
        cube([stone_side, stone_side, stone_side], center=true);
      }
    }
  };

// Final pyramid module
module pyramid() {
  pyramid_module();
}

// Call the pyramid module
pyramid();
