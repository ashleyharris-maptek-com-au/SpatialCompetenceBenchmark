// Tallest pyramid the pharoh can build with budgeted stones
// OpenSCAD model (units: meters, precision: millimetre)
// - Stone: 0.5 m (500 mm) cube
// - Stones available: 2,500,000
// - Layering rule: each upper-layer cube sits with its 4 bottom corners on centers of 4 cubes below
//   => stepped square pyramid with integer layers; base has N x N stones, next layer (N-1)x(N-1), ..., 1x1
// - No internal voids: the solid is the union of full layers
// Orientation: pyramid is centered at origin; +Y is north; tomb is at (0,0,0).
//
// Summary (computed below by echoes):
//   Maximum complete layers N = 195
//   Stones used = 2,490,670
//   Stones leftover = 9,330
//   Stone volume = 0.5^3 = 0.125 m^3
//   Total pyramid volume = 2,490,670 * 0.125 = 311333.75 m^3
//
// The geometry below models the pyramid as a stack of planar layers (each layer a single cuboid of thickness 0.5 m).
// This is numerically exact (each layer is an integer multiple of 0.5 m) and minimal (195 solids), so it is precise to the
// millimetre while remaining viewable/editable in typical CAD viewers.
//
// If the mason wants explicit exposed-stone grid plates for cutting guides, set show_grid = true (this will add thin plates
// aligned to each exposed stone; note that turning the grid on increases geometry complexity).

// ========= Parameters =========
s = 0.5;                      // stone side length in meters (500 mm)
stones_available = 2500000;   // available stones
N = 195;                      // computed maximum complete layers for given stone budget

// Derived quantities (exact arithmetic where possible)
total_stones = N * (N + 1) * (2 * N + 1) / 6;   // sum_{i=1..N} i^2
stones_used = total_stones;
stones_leftover = stones_available - stones_used;
volume_per_stone = s * s * s;
total_volume_m3 = stones_used * volume_per_stone;

// Echo values to the OpenSCAD console for verification
echo("stone_side_m", s);
echo("layers_N", N);
echo("stones_used", stones_used);
echo("stones_leftover", stones_leftover);
echo("total_volume_m3", total_volume_m3);

// ========= Geometry modules =========
module step_layer(size_in_stones, stone_size_m, z_base) {
    // place a single contiguous layer made of (size_in_stones x size_in_stones) stones
    layer_len = size_in_stones * stone_size_m;
    // Lower-left corner at (-layer_len/2, -layer_len/2) so layer is centered at origin in X/Y
    translate([-layer_len/2, -layer_len/2, z_base])
        cube([layer_len, layer_len, stone_size_m], center=false);
}

module stepped_pyramid(num_layers, stone_size_m) {
    union() {
        // layer index: 0 .. num_layers-1 (0 is bottom); each layer thickness = stone_size_m
        for (layer = [0 : num_layers - 1]) {
            size_in_stones = num_layers - layer;
            z = layer * stone_size_m;
            step_layer(size_in_stones, stone_size_m, z);
        }
    }
}

// Optional: thin plates showing exposed-stone grid outlines (for masonry cutting guides)
// Very lightweight visually (thin plates) but will increase element count; disabled by default.
show_grid = false;           // set to true to enable per-stone grid plates on exposed faces
grid_plate_thickness = 0.002; // 2 mm thick guide plates

module exposed_face_grid(num_layers, stone_size_m, thickness) {
    union() {
        for (layer = [0 : num_layers - 1]) {
            size = num_layers - layer;
            z_top = (layer + 1) * stone_size_m; // top surface of this layer
            // Top surface plate covering the entire layer (so stone subdivision lines are reconstructable)
            translate([-size*stone_size_m/2, -size*stone_size_m/2, z_top - thickness])
                cube([size*stone_size_m, size*stone_size_m, thickness], center=false);
            // Per-stone side guide plates along four faces (north/south/east/west)
            // North (positive Y) and South (negative Y) faces:
            for (i = [0 : size - 1]) {
                x_center = (i - (size - 1)/2) * stone_size_m;
                y_north = (size - 1)/2 * stone_size_m;
                y_south = -y_north;
                // north strip per stone (thin in Y)
                translate([x_center - stone_size_m/2, y_north, layer * stone_size_m])
                    cube([stone_size_m, thickness, stone_size_m], center=false);
                // south strip per stone
                translate([x_center - stone_size_m/2, y_south - thickness, layer * stone_size_m])
                    cube([stone_size_m, thickness, stone_size_m], center=false);
            }
            // East (positive X) and West (negative X) faces:
            for (j = [0 : size - 1]) {
                y_center = (j - (size - 1)/2) * stone_size_m;
                x_east = (size - 1)/2 * stone_size_m;
                x_west = -x_east;
                translate([x_east, y_center - stone_size_m/2, layer * stone_size_m])
                    cube([thickness, stone_size_m, stone_size_m], center=false);
                translate([x_west - thickness, y_center - stone_size_m/2, layer * stone_size_m])
                    cube([thickness, stone_size_m, stone_size_m], center=false);
            }
        }
    }
}

// ========= Render =========
// Stone color (visual only)
color([0.87, 0.72, 0.53]) stepped_pyramid(N, s);

// Optional grid overlays for the stonemason
if (show_grid) {
    color([0, 0, 0, 0.6]) exposed_face_grid(N, s, grid_plate_thickness);
}
