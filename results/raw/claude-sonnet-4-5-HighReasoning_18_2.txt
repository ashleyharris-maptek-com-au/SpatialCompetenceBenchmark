```openscad
stone_size = 500;
base_stones = 121;

module stepped_pyramid() {
    for (level = [0:base_stones-1]) {
        size = (base_stones - level) * stone_size;
        translate([0, 0, level * stone_size + stone_size/2])
            cube([size, size, stone_size], center=true);
    }
}

module marble_facade() {
    base = base_stones * stone_size;
    height = base_stones * stone_size;
    h = base / 2;
    
    polyhedron(
        points=[
            [-h,-h,0], [h,-h,0], [h,h,0], [-h,h,0],
            [0,0,height]
        ],
        faces=[
            [0,1,2,3], [0,4,1], [1,4,2], [2,4,3], [3,4,0]
        ]
    );
}

color("SandyBrown", 0.9) stepped_pyramid();
color("WhiteSmoke", 0.6) marble_facade();
```
