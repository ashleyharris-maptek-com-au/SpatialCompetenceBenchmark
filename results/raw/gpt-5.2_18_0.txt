// Tallest stable stepped pyramid from 20,000 cubes of side 0.5 m,
// with strict corner-support rule => each upper cube must be centered on a lower cube.
// This creates a 45° stepped pyramid: each higher layer shrinks by 1 cube on every side.
//
// Units: millimeters (SI)
// Orientation: North is +Y. Pyramid is centered over the tomb at the origin (0,0).
// Base sits on Z=0, pyramid grows in +Z.

$fn = 16;

cube_size = 500;          // mm (0.5 m)
total_stones = 20000;

// Tallest possible with perfect square layers:
// Total cubes for height H (layers 0..H-1): sum_{k=0}^{H-1} ( (2k+1)^2 ) = H(4H^2 - 1)/3
// Find max H with <= 20000
// H=24 => 24*(4*576-1)/3 = 24*2303/3 = 18424
// H=25 => 25*(4*625-1)/3 = 25*2499/3 = 20825 > 20000
height_layers = 24;
used_stones = height_layers * (4*height_layers*height_layers - 1) / 3; // 18424
leftover_stones = total_stones - used_stones;                          // 1576 (unused)

// Base layer side length in cubes: 2*H-1
base_n = 2*height_layers - 1; // 47 cubes per side
base_side_mm = base_n * cube_size; // 23500 mm

// Optimized: model as a single polyhedron with exact stepped profile (no internal voids).
// This is the exact union of all cubes under the corner-support constraint for H=24,
// with layers shrinking by 1 cube (500 mm) on each side per level.
//
// Construction: for each z step, the half-extent in x/y is:
// e(z) = base_side_mm/2 - cube_size * floor(z / cube_size)
// with z in [0, height_layers*cube_size]
//
// Create polyhedron by stacking rectangular rings per level as a stepped frustum with vertical faces.
// This matches the exact voxel union at millimeter precision.

module stepped_pyramid(H, s, n_base){
    // H: layers, s: cube size (mm), n_base: cubes per side at base (odd)
    // Total height:
    h_total = H * s;

    // Half-extent at base outer boundary:
    e0 = (n_base * s)/2;

    // Build vertices for each level boundary (z = k*s), k=0..H
    // At boundary k, plan extent equals e0 - k*s
    // Note: At top boundary k=H, extent is e0 - H*s = ( (2H-1)/2 - H )*s = 0.5*s
    // That corresponds to the outer boundary of the top cube.
    pts = [
        for (k = [0:H]) let(
            z = k*s,
            e = e0 - k*s
        )
        // 4 corners at this level (counterclockwise when viewed from +Z):
        [-e, -e, z],
        [ e, -e, z],
        [ e,  e, z],
        [-e,  e, z]
    ];

    // Faces: connect each band between level k and k+1 with 4 vertical trapezoid strips (split to triangles)
    // Plus bottom and top faces.
    faces = concat(
        // Side faces (each quad split into 2 triangles)
        [
            for (k = [0:H-1]) let(
                b = 4*k,       // base index for level k in pts
                t = 4*(k+1)    // base index for level k+1
            )
            // South face (-Y): pts b0-b1-t1-t0
            [b+0, b+1, t+1],
            [b+0, t+1, t+0],

            // East face (+X): b1-b2-t2-t1
            [b+1, b+2, t+2],
            [b+1, t+2, t+1],

            // North face (+Y): b2-b3-t3-t2
            [b+2, b+3, t+3],
            [b+2, t+3, t+2],

            // West face (-X): b3-b0-t0-t3
            [b+3, b+0, t+0],
            [b+3, t+0, t+3]
        ],

        // Bottom face (z=0), oriented outward (downward): two triangles
        [
            [0, 2, 1],
            [0, 3, 2]
        ],

        // Top face (z=H*s): last level indices start at 4*H
        let(top = 4*H)
        [
            [top+0, top+1, top+2],
            [top+0, top+2, top+3]
        ]
    );

    polyhedron(points = pts, faces = faces, convexity = 10);
}

// Centered on tomb at origin; north is +Y by convention, so no rotation needed.
stepped_pyramid(height_layers, cube_size, base_n);
