// Pyramid dimensions
side_length = 50; // cm
stone_volume = side_length^3; // cubic cm

// Total available volume
total_volume = 20000 * stone_volume; // cubic cm

// Calculate the height of the pyramid
height = 0;
while (true) {
    volume = (height * (height + 1) * (height + 2)) / 6 * side_length^2; // cubic cm
    if (volume > total_volume) {
        height -= 1;
        break;
    }
    height += 1;
}

// Calculate the number of layers
layers = height;

// OpenSCAD model
module stone(x, y, z) {
    translate([x, y, z]) cube(side_length, center=true);
}

module pyramid(layers, side_length) {
    for (layer = 0; layer < layers; layer++) {
        for (x = -layer; x <= layer; x++) {
            for (y = -layer; y <= layer; y++) {
                stone(x * side_length, y * side_length, layer * side_length);
            }
        }
    }
}

pyramid(layers, side_length);
