// Pharaoh's Pyramid Plan
// This OpenSCAD file generates a precise 3D model of the tallest stepped pyramid
// the pharaoh can build, adhering to the specified construction rules and stone budget.

// Units: All dimensions are in millimeters (mm).

// --- Constants ---
// Side length of a single cut stone cube (50cm = 0.5m)
stone_side_mm = 500;

// Maximum number of layers (N) for the pyramid.
// This is calculated based on the sum of squares formula for a pyramid
// where each layer's side length decreases by 1 stone, down to a 1x1 top stone.
// Total stones T = N * (N + 1) * (2*N + 1) / 6.
// With a budget of 20,000 stones:
// For N = 38: T = 38 * 39 * 77 / 6 = 19019 stones. This fits the budget.
// For N = 39: T = 39 * 40 * 79 / 6 = 20540 stones. This exceeds the budget.
num_layers = 38;

// Optional: For smoother visualization of any curved elements, though primarily
// dealing with cubes here. Set to a reasonable value for CAD viewing.
$fn = 32;

// --- Modules ---
// Module to define a single stone cube.
module stone() {
    cube(stone_side_mm);
}

// Main module to construct the entire stepped pyramid.
// It iterates through each layer and places the appropriate number of stones,
// adjusting for the required horizontal offset.
module pharaohs_pyramid() {
    // Iterate from the bottom layer (k=0) to the top layer (k=num_layers-1)
    for (k = [0 : num_layers - 1]) {
        
        // Determine the number of stones along one side of the current layer.
        // The base layer (k=0) has 'num_layers' stones per side,
        // and each subsequent layer has one less stone per side.
        current_layer_side_stones = num_layers - k;
        
        // Calculate the horizontal shift for the current layer.
        // The rule "each stone must have all 4 base corners sitting in the middle of another stone"
        // implies that each upper layer is offset inwards by half a stone's side length (stone_side_mm / 2)
        // in both the X and Y directions, relative to the grid of the layer below it.
        layer_horizontal_shift = k * (stone_side_mm / 2);
        
        // Calculate the starting coordinates (bottom-front-left corner) for the
        // current layer's stones, ensuring the entire pyramid is centered at (0,0,0) on the ground.
        // The total base dimension of the pyramid is num_layers * stone_side_mm.
        // The pyramid's overall bottom-front-left corner would be at (-(total_base_dim / 2), -(total_base_dim / 2), 0).
        // Each layer's starting position is offset from this overall base corner by 'layer_horizontal_shift'.
        
        total_base_dimension = num_layers * stone_side_mm;
        
        x_layer_start = -(total_base_dimension / 2) + layer_horizontal_shift;
        y_layer_start = -(total_base_dimension / 2) + layer_horizontal_shift;
        z_layer_start = k * stone_side_mm; // The height of the bottom of the current layer
        
        // Iterate through the grid of stones for the current layer (X and Y placement)
        for (i = [0 : current_layer_side_stones - 1]) {
            for (j = [0 : current_layer_side_stones - 1]) {
                // Translate to the specific position for each stone's bottom-front-left corner
                translate([x_layer_start + i * stone_side_mm,
                           y_layer_start + j * stone_side_mm,
                           z_layer_start]) {
                    stone(); // Place a single stone cube at this position
                }
            }
        }
    }
}

// --- Render the Pyramid ---
// Call the main module to generate the complete pyramid structure.
pharaohs_pyramid();
