// Parameters
stone_side_length = 50; // in mm
stones_available = 600000;
men_per_stone = 10;
total_men = 100000;

// Calculate the maximum number of stones that can be moved
stones_to_use = total_men / men_per_stone;

// Ensure we don't use more stones than available
stones_to_use = min(stones_to_use, stones_available);

// Calculate the dimensions of the pyramid
// We need to find the largest pyramid that can be built with the given stones
// The volume of the pyramid in terms of stones is (base_area * height) / stone_volume
// The base area is a square, so base_area = side_length^2
// The height is the number of stones stacked vertically

// Volume of one stone
stone_volume = stone_side_length^3;

// We need to find the largest pyramid that fits within the number of stones available
// Let's assume the pyramid is a right square pyramid
// The volume of the pyramid is (base_area * height) / 3

// We need to find the largest integer side_length and height that fit within the number of stones
// Let's start with the height and decrease it until we find a valid configuration

function isValidPyramid(side_length, height) = 
    (side_length^2 * height) <= stones_to_use * 3;

function findLargestPyramid() =
    for (height = [stones_to_use; 1; -1]) 
        for (side_length = [int(sqrt(stones_to_use * 3 / height)); 1; -1]) 
            if (isValidPyramid(side_length, height)) 
                return [side_length, height];

pyramid_dimensions = findLargestPyramid();
side_length = pyramid_dimensions[0];
height = pyramid_dimensions[1];

// Now we need to generate the OpenSCAD code for the pyramid
module stone(x, y, z) {
    cube([stone_side_length, stone_side_length, stone_side_length], center=true);
}

module pyramid() {
    for (z = [0:height-1]) 
        for (x = [-side_length/2:side_length/2-1]) 
            for (y = [-side_length/2:side_length/2-1]) 
                if (x*x + y*y <= (side_length/2 - z)*(side_length/2 - z)) 
                    stone(x * stone_side_length, y * stone_side_length, z * stone_side_length);
}

// Rotate the pyramid to face north (+y)
rotate([90, 0, 0]) 
pyramid();
