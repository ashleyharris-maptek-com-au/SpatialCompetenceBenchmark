// OpenSCAD plan: tallest pyramid the pharaoh can build with 150,000 stones (50 cm cubes)
// All dimensions are in millimeters (mm)

N = 76;                 // number of layers (maximum so that sum_{k=1..N} k^2 <= 150000)
STONE_SIDE = 500;        // side length of each cube in mm (50 cm)
LAYER_THICKNESS = 500;    // thickness of each layer in mm (equal to stone height)

// Build the pyramid as a stack of square prisms (each layer is a solid slab of (N - k) x (N - k) stones)
module pyramid() {
    union() {
        for (k = [0 : N - 1]) {
            L = (N - k) * STONE_SIDE;                 // side length of this layer in mm
            translate([k * STONE_SIDE / 2, k * STONE_SIDE / 2, k * LAYER_THICKNESS])
                cube([L, L, LAYER_THICKNESS], center = false);
        }
    }
}

// Optional: print summary to console for verification
echo("Pyramid layers: ", N);
echo("Base side (mm): ", N * STONE_SIDE);
echo("Total height (mm): ", N * LAYER_THICKNESS);
echo("Total approximate volume (mm^3): ", 0);

// Render the pyramid
pyramid();
