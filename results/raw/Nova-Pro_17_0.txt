// Conversion factors
yd_to_m = 0.9144; // 1 yard to meters
ft_to_m = 0.3048; // 1 foot to meters
gal_to_m3 = 0.00454609; // 1 UK gallon to cubic meters

// Initial volume in cubic meters
initial_volume_m3 = 5 * (yd_to_m ^ 3);

// Volume added from trucks in cubic meters
truck_volume_m3 = 1500 * (ft_to_m ^ 3);

// Volume added from train in cubic meters
train_volume_m3 = 100;

// Volume added from drums in cubic meters
drum_volume_m3 = 20 * 44 * gal_to_m3;

// Total volume in cubic meters
total_volume_m3 = initial_volume_m3 + truck_volume_m3 + train_volume_m3 + drum_volume_m3;

// Diameter and radius of the silo in meters
silo_diameter_m = 10 * yd_to_m;
silo_radius_m = silo_diameter_m / 2;

// Angle of repose in radians
angle_of_repose_rad = 33 * (pi / 180);

// Height of the cone-shaped pile
height = (total_volume_m3 / (pi * silo_radius_m ^ 2)) * tan(angle_of_repose_rad);

// OpenSCAD code to create the sand pile
module sand_pile() {
  difference() {
    cylinder(h = height, r = silo_radius_m);
    translate([0, 0, height]) rotate([angle_of_repose_rad * (180 / pi), 0, 0]) cylinder(h = silo_radius_m, r = silo_radius_m);
  }
}

sand_pile();
