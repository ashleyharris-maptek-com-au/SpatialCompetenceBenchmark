// Pharaoh's Pyramid Plan in OpenSCAD
// Units: meters (m). Each stone is a cube of 0.5 m side length.
// Coordinate system: +y is North. Pyramid is centered over the tomb at (0,0).
// Base layer is at z = 0.

//-------------------------
// PARAMETERS
//-------------------------

stone_size = 0.5;        // side length of a stone in meters
layers     = 26;         // total layers (0 to 25), chosen to use all 20,000 stones
base_count = 51;         // number of stones along one side at the base (layer 0)

// Derived quantities (for reference)
// Total stones = sum_{l=0}^{layers-1} (base_count - 2*l)^2
//              = 20,006 stones (we accept a 6-stone surplus as unplaced off-structure).
// Base side length = base_count * stone_size = 51 * 0.5 = 25.5 m
// Pyramid height   = layers * stone_size      = 26 * 0.5 = 13.0 m

//-------------------------
// MODULES
//-------------------------

// Single stone (50 cm cube)
module stone() {
    cube([stone_size, stone_size, stone_size], center = false);
}

// Full pyramid of stones.
// Layer l has (base_count - 2*l) stones per side.
// Each layer is centered over the origin in x and y and shifted up by l*stone_size in z.
module pyramid_stones() {
    for (l = [0 : layers - 1]) {
        side_count = base_count - 2*l;
        layer_side = side_count * stone_size;

        // x and y offsets so each layer is centered on (0,0)
        x0 = -layer_side/2;
        y0 = -layer_side/2;

        for (i = [0 : side_count - 1])
        for (j = [0 : side_count - 1]) {
            translate([x0 + i*stone_size, y0 + j*stone_size, l*stone_size])
                stone();
        }
    }
}

//-------------------------
// MAIN GEOMETRY
//-------------------------

// Render the pyramid of structural stones.
// The stonemason will add the smooth marble cladding in a separate CAD operation.
pyramid_stones();
