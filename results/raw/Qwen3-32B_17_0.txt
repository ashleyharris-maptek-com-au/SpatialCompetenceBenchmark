// Convert all volumes to cubic meters
// 5 cubic yards = 5 * 0.764555 = 3.822775 m³
// 1500 cubic feet = 1500 * 0.0283168 = 42.4752 m³
// 100 cubic meters = 100 m³
// 20 drums of 44 UK gallons = 20 * 44 * 0.00454609 = 4.0 m³ (approx)

total_vol = 3.822775 + 42.4752 + 100 + 4.0;

// Silo diameter in meters: 10 yards = 9.144 meters
diameter = 9.144;
radius = diameter / 2;

// Angle of repose in radians
theta = 33 * pi / 180;

// Volume of a cone: V = (1/3) * pi * r² * h
// We can solve for h: h = 3V / (pi * r²)
height = 3 * total_vol / (pi * radius * radius);

// The base radius of the pile is: r_pile = h / tan(theta)
r_pile = height / tan(theta);

// Limit the pile to the silo radius
r_pile = min(r_pile, radius);

// Create the conical pile
translate([0, 0, 0]) {
  cylinder(h = height, r1 = r_pile, r2 = 0, center = false);
}
