// Convert all volumes to cubic meters
// 5 cubic yards = 3.823 m³
// 1500 cubic feet = 42.5 m³
// 100 cubic meters = 100 m³
// 20 drums @ 44 UK gallons = 20 * 200.2 = 4004 liters = 4.004 m³

// Total volume = 3.823 + 42.5 + 100 + 4.004 = 150.327 m³

// Silo diameter = 10 yards = 9.144 meters
// Radius R = 9.144 / 2 = 4.572

// Angle of repose 33 degrees
// The pile forms a cone inside the silo
// V = (1/3) * π * r² * h
// h = r * tan(angle)

// Let r = min(silo_radius, r_cone)
// Solve for r_cone such that volume is 150.327 m³

// We assume the pile is a cone within the silo
// If the cone radius exceeds the silo radius, it's truncated
// For simplicity, we calculate the cone height and radius based on the volume and angle

// Calculate cone radius and height
angle = 33;
R_silo = 4.572;
V = 150.327;

// h = r * tan(angle)
// V = (1/3) * pi * r^2 * h = (1/3) * pi * r^3 * tan(angle)
// Solve for r:
r = (V * 3 / (pi * tan(angle)))^(1/3);
h = r * tan(angle);

// If r exceeds silo radius, truncate the cone
if (r > R_silo) {
    // The cone touches the silo wall at radius R_silo
    // Compute new height based on angle
    h = R_silo * tan(angle);
    // Compute volume of the truncated cone
    V = (1/3) * pi * R_silo^2 * h;
    // If volume is less than needed, it's a frustum
    // But for simplicity, we will only model the cone within the silo
}

difference() {
    cylinder(h = h, r1 = R_silo, r2 = 0, center = false);
    translate([0,0,-1]) cube([2*R_silo, 2*R_silo, h + 1], center = true);
}
