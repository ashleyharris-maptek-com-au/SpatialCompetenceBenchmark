// OpenSCAD plan: Tallest pyramid plan using cubic stones of 500 mm side
// Each stone requires 10 men to move; available army: 100,000 men
// Budget: 2,500,000 stones total, but movement constraint caps to max_stones_by_men
// With 100,000 / 10 = 10,000 stones movable total, the largest N satisfies sum_{i=1..N} i^2 <= 10,000
// Stone size (mm)
stone = 500; // cubic stone side length in mm

// Manpower constraint
manpower = 100000;
move_per_stone = 10;
max_stones_by_men = floor(manpower / move_per_stone); // 10000 stones

// Compute max N such that total stones sum i^2 <= max_stones_by_men
N = 0;
sum_stones = 0;
while (sum_stones + (N + 1)*(N + 1) <= max_stones_by_men) {
    N = N + 1;
    sum_stones = sum_stones + N*N;
}

// Derived dimensions (mm)
base_width = N * stone; // width of base square
pyramid_height = N * stone;

// Display computed results (for verification in the CAD console)
echo("Max layers per side (N): ", N);
echo("Total stones used (sum i^2): ", sum_stones);
echo("Base width (mm): ", base_width);
echo("Pyramid height (mm): ", pyramid_height);
echo("Total stone volume (m^3): ", sum_stones * 0.125);

// Build the pyramid: layers from bottom (layer=0) to top (layer=N-1)
// Each layer k has (N - k) x (N - k) stones, arranged centered, with z = k * stone
module pyramid() {
    for (layer = [0 : N - 1]) {
        n = N - layer;                // stones per side in this layer
        z = layer * stone;            // height of this layer
        layer_width = n * stone;      // width/length of this layer
        // Place stones on a square grid centered at origin
        for (i = [0 : n - 1]) {
            x = -layer_width/2 + i * stone;
            for (j = [0 : n - 1]) {
                y = -layer_width/2 + j * stone;
                translate([x, y, z])
                    cube([stone, stone, stone], center = false);
            }
        }
    }
}

// Render the pyramid
pyramid();
