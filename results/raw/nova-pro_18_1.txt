```openscad
// Configuration parameters
STONE_SIDE_LENGTH = 50; // in mm
STONES_AVAILABLE = 150000;
MEN_AVAILABLE = 100000;
MEN_PER_STONE = 10;

// Calculate the maximum number of stones that can be moved
STONES_THAT_CAN_BE_MOVED = MEN_AVAILABLE / MEN_PER_STONE;

// Ensure we do not exceed the available stones
NUM_STONES_TO_USE = min(STONES_THAT_CAN_BE_MOVED, STONES_AVAILABLE);

// Function to calculate the height of a pyramid given the number of stones
function pyramid_height(n) = 
  let (layers = ceil(n^(1/3))) in
    layers * STONE_SIDE_LENGTH;

// Calculate the number of stones needed for a pyramid of given height
function stones_needed_for_height(h) = 
  let (layers = h / STONE_SIDE_LENGTH) in
    layers^3;

// Binary search to find the tallest pyramid height
function binary_search_height(low, high) =
  if (high <= low)
    low
  else
    let (mid = (low + high) / 2) in
      if (stones_needed_for_height(mid) <= NUM_STONES_TO_USE)
        binary_search_height(mid, high)
      else
        binary_search_height(low, mid);

// Find the maximum height of the pyramid
MAX_HEIGHT = binary_search_height(0, NUM_STONES_TO_USE);

// Convert height to number of layers
NUM_LAYERS = MAX_HEIGHT / STONE_SIDE_LENGTH;

// OpenSCAD module to create the pyramid
module pyramid() {
  for (layer = [0 : NUM_LAYERS - 1]) {
    let (stones_in_layer = (NUM_LAYERS - layer)^2) in
      for (stone = [0 : stones_in_layer - 1]) {
        let (row = floor(stone / (NUM_LAYERS - layer))) in
          let (col = stone % (NUM_LAYERS - layer)) in
            translate([
              col * STONE_SIDE_LENGTH,
              layer * STONE_SIDE_LENGTH,
              row * STONE_SIDE_LENGTH
            ])
              cube([STONE_SIDE_LENGTH, STONE_SIDE_LENGTH, STONE_SIDE_LENGTH]);
      }
  }
}

// Generate the pyramid
pyramid();
```
