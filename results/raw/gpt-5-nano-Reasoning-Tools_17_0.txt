$fn = 60;
R = 4.572;
V = 150.29860337792; // total sand volume in m^3
t = tan(33 * pi / 180); // tan(alpha) with alpha=33°
z = (V + (2.0/3.0) * pi * R * R * R * t) / (pi * R * R) - R * t;
H = R * t; // height of the top cone
cylinder(h = z, r = R, center = false);
translate([0,0,z]) cylinder(h = H, r1 = R, r2 = 0, center = false);
