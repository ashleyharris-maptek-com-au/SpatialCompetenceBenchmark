```scad
// Define the side length of a stone
stone_side = 50; // in millimeters

// Calculate the maximum number of stones that can be used
total_stones = 20000;

// Calculate the maximum height of the pyramid
max_height = total_stones / 4; // because each layer uses 4 times the stones of the previous layer

// Create the pyramid structure
module pyramid(level, current_height, current_size) {
    if (level == 0)
        return;

    size = current_size;
    height = current_height;
    
    for (x = [-size/2:size/2]) {
        for (y = [-size/2:size/2]) {
            translate([x*stone_side, y*stone_side, height*stone_side])
                cube([stone_side, stone_side, stone_side], center = true);
        }
    }
    
    pyramid(level - 1, height - stone_side/2, size - stone_side);
}

// Generate the pyramid with maximum height
pyramid(max_height, max_height*stone_side/2, max_height*stone_side);
```
