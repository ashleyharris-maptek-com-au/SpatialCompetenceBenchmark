// OpenSCAD plan: Tallest pyramid given manpower and 0.5 m cubic stones.
// Each stone rests on four beneath; base is N x N with layers reducing by 1 per level.
// Manpower: 100,000 men, 10 per stone => 10,000 stones max. With 0.5 m cubes,
// the largest full pyramid that fits is N = 30 (sum k^2 from k=1..30 = 9,455 stones).

STONE = 500;          // mm side length of each cube (0.5 m)
N = 30;                 // base stones per side (30 x 30 on bottom layer)

TOTAL_STONES = N*(N+1)*(2*N+1)/6;           // 9,455 stones
TOTAL_VOL_MM3 = TOTAL_STONES * STONE * STONE * STONE; // mm^3
TOTAL_VOL_M3 = TOTAL_VOL_MM3 / 1e9;          // m^3

// Debug/plan metadata (printed to console)
echo("Total stones in pyramid:", TOTAL_STONES);
echo("Total volume (mm^3):", TOTAL_VOL_MM3);
echo("Total volume (m^3):", TOTAL_VOL_M3);

module Stone() {
    cube([STONE, STONE, STONE], center = true);
}

// Build pyramid layer by layer:
// z = 0 is bottom layer; z increases upward.
// Each layer z has n = N - z stones per side, arranged so that top stones sit on 4 below.
for (z = [0 : N-1]) {
    n = N - z;
    offset = z * 0.5;              // layer offset in stone units (0.5 indicates half-stone shifts)
    zpos = z * STONE + STONE/2;    // center height of this layer (mm)

    for (i = [0 : n-1]) {
        x = (i + offset + 0.5) * STONE; // center x-position of stone in this row
        for (j = [0 : n-1]) {
            y = (j + offset + 0.5) * STONE; // center y-position of stone in this column
            translate([x, y, zpos])
                Stone();
        }
    }
}
