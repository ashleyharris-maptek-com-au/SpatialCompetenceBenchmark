def solve_shikaku(grid):
    rows = len(grid)
    cols = len(grid[0])
    clues = {}
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] != 0:
                clues[(c, r)] = grid[r][c]

    regions = []
    filled = [[False for _ in range(cols)] for _ in range(rows)]

    def is_valid(r, c):
        return 0 <= r < rows and 0 <= c < cols

    def find_empty():
        for r in range(rows):
            for c in range(cols):
                if not filled[r][c]:
                    return r, c
        return None

    def get_possible_rectangles(r, c, area):
        possible = []
        # Iterate over all possible bottom-right corners
        for r2 in range(r, rows):
            for c2 in range(c, cols):
                width = c2 - c + 1
                height = r2 - r + 1
                current_area = width * height

                if current_area == area:
                    # Check if this rectangle contains only empty cells (except the clue)
                    is_clue_inside = False
                    contains_other_clue = False
                    for rr in range(r, r2 + 1):
                        for cc in range(c, c2 + 1):
                            if (cc, rr) in clues:
                                if (cc, rr) == (c, r):
                                    is_clue_inside = True
                                else:
                                    contains_other_clue = True
                                    break
                        if contains_other_clue:
                            break
                    
                    if is_clue_inside and not contains_other_clue:
                        possible.append((c, r, c2, r2))
        return possible

    def solve():
        empty_cell = find_empty()
        if not empty_cell:
            return True # All cells filled

    rows = len(grid_values)
    cols = len(grid_values[0])
    clues = {}

                clues[(c, r)] = grid_values[r][c]

    regions = []
    filled = [[False for _ in range(cols)] for _ in range(rows)]

    def find_empty_cell():

    def get_valid_rectangles_for_clue(clue_c, clue_r, clue_area):

                for r2 in range(r1, rows):

                        width = c2 - c1 + 1
                        height = r2 - r1 + 1
                        area = width * height

                        if area == clue_area:

import sys

# Increase recursion depth for deep backtracking

def solve_shikaku(grid_values):
    rows = len(grid_values)
    cols = len(grid_values[0])
    clues = {}
    for r in range(rows):
        for c in range(cols):
            if grid_values[r][c] != 0:
                clues[(c, r)] = grid_values[r][c]

    regions = []
    filled = [[False for _ in range(cols)] for _ in range(rows)]

    def find_empty_cell():
        for r in range(rows):
            for c in range(cols):
                if not filled[r][c]:
                    return r, c
        return None

    def solve():
        empty_cell = find_empty_cell()
        if not empty_cell:
            return True # Solved

        r, c = empty_cell

        # Try all possible rectangles that COVER (r, c)
        # A rectangle is defined by its top-left (c1, r1) and bottom-right (c2, r2) corners.
        # So, we need c1 <= c <= c2 and r1 <= r <= r2.

        # Iterate over all possible top-left corners (c1, r1) such that c1 <= c and r1 <= r
        for r1 in range(r + 1):
            for c1 in range(c + 1):
                # Iterate over all possible bottom-right corners (c2, r2) such that c2 >= c and r2 >= r
                for r2 in range(r, rows):
                    for c2 in range(c, cols):
                        
                        width = c2 - c1 + 1
                        height = r2 - r1 + 1
                        area = width * height

                        # Check if this potential rectangle is entirely empty so far
                        is_empty_and_valid_shape = True
                        for rr in range(r1, r2 + 1):
                            for cc in range(c1, c2 + 1):
                                if filled[rr][cc]:
                                    is_empty_and_valid_shape = False
                                    break
                            if not is_empty_and_valid_shape:
                                break
                        
                        if not is_empty_and_valid_shape:
                            continue

                        # Now check for clues within this rectangle [c1, r1, c2, r2]
                        clue_in_rect_loc = None
                        clue_count = 0
                        
                        for clue_loc, clue_val in clues.items():
                            clue_c, clue_r = clue_loc
                            if c1 <= clue_c <= c2 and r1 <= clue_r <= r2:
                                clue_count += 1
                                if clue_val == area:
                                    clue_in_rect_loc = clue_loc
                                else: # Clue value doesn't match area
                                    clue_count = 2 # Mark as invalid (more than one clue, or a clue with wrong area)
                                    break
                        
                        if clue_count == 1 and clue_in_rect_loc:
                            # This is a valid rectangle for clue_in_rect_loc
                            # Place it
                            for rr in range(r1, r2 + 1):
                                for cc in range(c1, c2 + 1):
                                    filled[rr][cc] = True
                            regions.append((c1, r1, c2, r2))

                            if solve():
                                return True

                            # Backtrack
                            regions.pop()
                            for rr in range(r1, r2 + 1):
                                for cc in range(c1, c2 + 1):
                                    filled[rr][cc] = False
        
        # If no rectangle covering (r, c) leads to a solution
        return False

    if solve():
        # Sort regions for consistency, e.g., by top-left corner
        regions.sort(key=lambda x: (x[1], x[0]))
        return regions
    else:
        return None
