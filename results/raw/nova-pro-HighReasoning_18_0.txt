// Define the size of each stone
stone_size = 50; // in mm

// Calculate the number of stones available per layer
stones_per_layer = 20000;

// Function to calculate the number of stones in a layer given its size
function stones_in_layer(layer_size) = (layer_size * layer_size);

// Initialize variables
max_height = 0;
current_stones = stones_per_layer;

// Find the maximum height of the pyramid
for (height = [1:int(sqrt(stones_per_layer))]) {
    stones_needed = 0;
    for (layer = [1:height]) {
        layer_size = height - layer + 1;
        stones_needed += stones_in_layer(layer_size);
    }
    if (stones_needed <= stones_per_layer) {
        max_height = height;
        current_stones -= stones_needed;
    } else {
        break;
    }
}

// OpenSCAD module to generate the pyramid
module pyramid() {
    for (layer = [1:max_height]) {
        layer_size = max_height - layer + 1;
        translate([layer_size * -stone_size / 2, 0, (max_height - layer) * stone_size])
        for (x = [0:layer_size - 1]) {
            for (z = [0:layer_size - 1]) {
                translate([x * stone_size, 0, z * stone_size])
                cube([stone_size, stone_size, stone_size]);
            }
        }
    }
}

// Generate the pyramid
pyramid();
