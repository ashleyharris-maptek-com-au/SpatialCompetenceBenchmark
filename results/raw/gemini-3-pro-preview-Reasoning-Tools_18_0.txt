// Pharaoh's Tomb Pyramid Construction Plan
// ---------------------------------------------------------
// Design Parameters & Constraints Analysis:
//
// 1. Labor Constraint:
//    - Total Army: 100,000 men
//    - Labor Cost: 10 men per stone
//    - Transport Capacity: 100,000 / 10 = 10,000 stones max.
//
// 2. Material Constraint:
//    - Inventory: 20,000 stones.
//    - Effective Constraint: min(20,000, 10,000) = 10,000 stones.
//
// 3. Geometry & Structural Integrity:
//    - Stone Size: 0.5m x 0.5m x 0.5m cube.
//    - Stability Rule: Corners must sit on centers of stones below.
//    - This mandates a uniform setback of 1/2 stone width (0.25m) per layer.
//    - This forms a square pyramidal core structure.
//    - Stone Count for H layers = Sum(k^2) for k=1 to H.
//
// 4. Maximum Height Calculation:
//    - We need max H where Sum(k^2) <= 10,000.
//    - For H=30: 9,455 stones.
//    - For H=31: 10,416 stones (Exceeds labor budget).
//    - Result: Max Height = 30 layers (15 meters).
//
// 5. Facade Geometry (Metric SI):
//    - The marble facade must smooth the steps and cover every stone.
//    - To minimize material while covering, the facade plane connects the 
//      top-outer corners of the bottom step and the top step.
//    - Step Logic: 
//      - Rise per step = 0.5m.
//      - Run per step = 0.25m.
//      - Slope = 2.0.
//    - Apex Calculation:
//      - Top Stone (Layer 30) Top Face: Z = 15.0m, Width = 0.5m.
//      - Facade extends from corner (x=0.25, z=15.0) to Apex (x=0).
//      - dx = 0.25, Slope = 2.0 -> dz = 0.5.
//      - Apex Height = 15.5m.
//    - Base Calculation:
//      - Bottom Stone (Layer 1) Top Face: Z = 0.5m, Width = 15.0m (x=7.5).
//      - Facade extends from corner (x=7.5, z=0.5) to Ground (z=0).
//      - dz = 0.5, Slope = 2.0 -> dx = 0.25.
//      - Base Edge = 7.75m. Total Base Width = 15.5m.

$fn = 100; // High resolution for the stone mason

// --- Dimensions (Meters) ---
stone_side = 0.500;
num_layers = 30;

// Facade dimensions derived from structural logic
facade_base_width = 15.500;
facade_height = 15.500;

// --- Modules ---

module stone_core() {
    // Renders the structural core of the pyramid.
    // Optimized: Uses 1 solid prism per layer instead of individual blocks
    // to improve viewport performance while maintaining exact volume boundaries.
    
    color("SandyBrown")
    for (i = [0 : num_layers - 1]) {
        // Layer calculation:
        // i=0 is bottom layer (index 0), width 30 stones
        // i=29 is top layer (index 29), width 1 stone
        
        stones_count = num_layers - i;
        layer_width_m = stones_count * stone_side;
        
        // Z position: Each layer is stacked on the previous
        z_pos = i * stone_side;
        
        translate([0, 0, z_pos + stone_side/2]) {
            cube([layer_width_m, layer_width_m, stone_side], center=true);
        }
    }
}

module marble_facade() {
    // Generates the marble wedges needed to create a smooth surface.
    // Defined as the Smooth Pyramid Volume minus the Structural Core Volume.
    
    color("White", 0.9) // White marble, slightly transparent to show core in preview
    difference() {
        // 1. Ideal Smooth Pyramid Geometry
        // Centered on X/Y, base at Z=0
        translate([0, 0, 0])
            linear_extrude(height = facade_height, scale = 0)
            square(facade_base_width, center = true);
        
        // 2. Subtract the core to leave only the facade wedges
        stone_core();
    }
}

// --- Assembly ---

// Grouping the assembly
union() {
    // 1. Place the structural stones
    stone_core();

    // 2. Place the marble cladding
    marble_facade();
}

// --- Information Echo ---
echo("------------------------------------------------");
echo("PYRAMID CONSTRUCTION PLAN");
echo("Total Layers:", num_layers);
echo("Total Height (m):", num_layers * stone_side);
echo("Base Width (m):", num_layers * stone_side);
echo("Total Stones Used:", 9455);
echo("Labor Units Used:", 94550);
echo("Facade Apex Height (m):", facade_height);
echo("Facade Base Width (m):", facade_base_width);
echo("------------------------------------------------");
