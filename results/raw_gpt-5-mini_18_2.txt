// Pharaoh's pyramid (OpenSCAD)
// Units: meters (SI)
// Stone cube side length: 0.5 m (500 mm)
// Available stones used for optimal tallest stepped pyramid: 597861 (N = 121 layers, base 121x121)
// External stones (visible faces, to be covered by marble): 29041
// This model includes only the external large stones (each stone modelled precisely as a 0.5 m cube, positioned to millimetre precision).
// Pyramid faces oriented such that +Y is north. Pyramid is centered on the origin (tomb at [0,0,0]).
// Note: model uses a stepped-stone stacking rule where each stone above sits with its four base corners on the centers of a 2x2 block below.

// Parameters
stoneside = 0.5;  // side length of a stone in meters
N = 121;          // base has N x N stones; number of layers = N

module stone() {
    // single stone, centered at the given translation point
    cube([stoneside, stoneside, stoneside], center=true);
}

// Build only the external stones (perimeter of each layer). This reduces geometry while keeping every
// external stone precise for the stonemason to cut marble facings.
for (k = [0 : N-1])              // k = 0 is base layer; k increases upward
    let (
        n = N - k,               // number of stones per side on this layer
        z = stoneside/2 + k*stoneside // center z coordinate for this layer's stones
    )
    for (i = [0 : n-1])
        for (j = [0 : n-1])
            // include stone if it's on the perimeter of this layer (or the single top stone)
            if (n == 1 || i == 0 || i == n-1 || j == 0 || j == n-1)
                translate([
                    (i - (n-1)/2) * stoneside, // x position (meters)
                    (j - (n-1)/2) * stoneside, // y position (meters); +Y is north
                    z
                ])
                    stone();
