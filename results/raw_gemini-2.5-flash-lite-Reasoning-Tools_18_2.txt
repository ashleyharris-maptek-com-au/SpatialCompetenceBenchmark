```openscad
// Units: Millimeters
// Stone size: 500mm x 500mm x 500mm

$fn = 100; // Higher resolution for rendering (optional for cubes)

// --- Configuration ---
stone_size = 500; // mm

// Based on the stone budget of 600,000 stones, the tallest pyramid
// has 121 layers.
// Total stones for N=121 layers: 1^2 + 2^2 + ... + 121^2 = 596851 stones.
// This leaves 600,000 - 596851 = 3149 stones.
total_layers = 121;

// --- Pyramid Module ---
// This module generates the pyramid layer by layer.
// It accounts for the construction rule: "each stone must have all 4 base
// corners sitting in the middle of another stone." This implies an alternating
// grid shift between layers.
module pyramid(total_layers, stone_size) {
    // Iterate through each layer, from bottom (layer 0) to top (layer total_layers - 1).
    for (l = [0 : total_layers - 1]) {
        // The number of stones on the side of the current layer.
        // The base layer (l=0) has total_layers stones on its side.
        // The top layer (l=total_layers-1) has 1 stone on its side.
        num_stones_on_side = total_layers - l;

        // The z-coordinate for the bottom face of the current layer.
        z_base = l * stone_size;

        // Determine the grid shift for this layer to satisfy the construction rule.
        // Layers with an even index (0, 2, 4...) are aligned on one grid.
        // Layers with an odd index (1, 3, 5...) are shifted by half a stone's width/depth.
        layer_grid_shift_x = 0;
        layer_grid_shift_y = 0;
        if (l % 2 != 0) { // Odd layers are shifted
            layer_grid_shift_x = 0.5 * stone_size;
            layer_grid_shift_y = 0.5 * stone_size;
        }

        // Calculate the centering offset for the current layer's grid of stones.
        // This aligns the geometric center of the s x s block of stones within this layer
        // to the pyramid's central axis (0,0).
        // The range of stone centers within a layer's intrinsic grid goes from 0 to (s-1)*stone_size.
        // The midpoint of this range is (s-1)/2.0 * stone_size. We shift by the negative of this value.
        current_layer_centering_offset_x = (num_stones_on_side - 1) / 2.0 * stone_size;
        current_layer_centering_offset_y = (num_stones_on_side - 1) / 2.0 * stone_size;

        // Place each stone in the current layer.
        // 'i' and 'j' are the 0-based indices for the stones within the current layer's grid.
        for (i = [0 : num_stones_on_side - 1]) {
            for (j = [0 : num_stones_on_side - 1]) {
                // Calculate the base position of the stone within its layer's internal,
                // unshifted grid. These coordinates are aligned with the start of the grid (0,0).
                local_x = i * stone_size;
                local_y = j * stone_size;

                // Apply the layer-specific grid shift (alternating pattern).
                // This positions the stone relative to a universal grid, accounting for the
                // "middle of another stone" rule, ensuring adjacent layers interlock correctly.
                shifted_x = local_x + layer_grid_shift_x;
                shifted_y = local_y + layer_grid_shift_y;

                // Apply the centering offset for this specific layer. This aligns the
                // block of 'num_stones_on_side' x 'num_stones_on_side' stones to the pyramid's
                // central X/Y axis (0,0).
                final_x = shifted_x - current_layer_centering_offset_x;
                final_y = shifted_y - current_layer_centering_offset_y;

                // The z position is the bottom face of the current layer.
                final_z = z_base;

                // Translate and draw the stone cube.
                // The 'cube()' primitive is defined from [0,0,0] to [stone_size, stone_size, stone_size].
                // The 'translate()' command moves its origin to [final_x, final_y, final_z].
                // This setup ensures the pyramid is centered over the tomb (assumed to be at 0,0,0)
                // and oriented with +y facing North.
                translate([final_x, final_y, final_z]) {
                    cube([stone_size, stone_size, stone_size]);
                }
            }
        }
    }
}

// --- Generate the Pyramid ---
// Call the module to create the pyramid using the calculated parameters.
pyramid(total_layers, stone_size);
```
