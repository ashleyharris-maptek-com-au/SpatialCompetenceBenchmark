def convex_hull(points):
    """
    Compute the convex hull of a set of 2D points.
    Returns indices of points on the convex hull in counter-clockwise order.
    """
    n = len(points)
    if n < 3:
        return list(range(n))
    
    # Create list of (point, index) tuples
    indexed_points = [(points[i], i) for i in range(n)]
    
    # Sort points lexicographically (first by x, then by y)
    indexed_points.sort()
    
    # Cross product of vectors OA and OB where O is origin
    def cross(o, a, b):
        return (a[0] - o[0]) * (b[1] - o[1]) - (a[1] - o[1]) * (b[0] - o[0])
    
    # Build lower hull
    lower = []
    for point, idx in indexed_points:
        while len(lower) >= 2 and cross(lower[-2][0], lower[-1][0], point) <= 0:
            lower.pop()
        lower.append((point, idx))
    
    # Build upper hull
    upper = []
    for point, idx in reversed(indexed_points):
        while len(upper) >= 2 and cross(upper[-2][0], upper[-1][0], point) <= 0:
            upper.pop()
        upper.append((point, idx))
    
    # Remove last point of each half because it's repeated
    lower.pop()
    upper.pop()
    
    # Concatenate lower and upper hull
    hull = lower + upper
    
    # Extract indices
    hull_indices = [idx for point, idx in hull]
    
    return hull_indices
