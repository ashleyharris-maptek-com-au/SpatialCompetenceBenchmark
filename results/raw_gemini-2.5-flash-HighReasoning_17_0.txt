$fn = 100;
R_silo = 10 * 0.9144 / 2;
theta_rad = 33 * PI / 180;
V_total = (5 * pow(0.9144, 3)) + (1500 * pow(0.3048, 3)) + 100 + (20 * 44 * 4.54609 / 1000);
h_cone = R_silo * tan(theta_rad);
V_cone = (1/3) * PI * pow(R_silo, 2) * h_cone;
h_cyl = (V_total - V_cone) / (PI * pow(R_silo, 2));
union() {
    cylinder(h = h_cyl, r = R_silo, center = false);
    translate([0, 0, h_cyl]) {
        cylinder(h = h_cone, r1 = R_silo, r2 = 0, center = false);
    }
}
