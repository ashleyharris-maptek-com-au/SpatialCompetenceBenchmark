```python
def convex_hull(points):
    """
    Computes the convex hull of a set of 2D points.
    Returns the indices of points on the hull in counter-clockwise order.
    Uses Andrew's monotone chain algorithm.
    """
    n = len(points)
    if n < 3:
        return list(range(n))
    
    # Create list of (point, index) and sort by x, then by y
    indexed_points = [(points[i], i) for i in range(n)]
    indexed_points.sort()
    
    # Build lower hull
    lower = []
    for point, idx in indexed_points:
        while len(lower) >= 2 and cross(points[lower[-2]], points[lower[-1]], point) <= 0:
            lower.pop()
        lower.append(idx)
    
    # Build upper hull
    upper = []
    for point, idx in reversed(indexed_points):
        while len(upper) >= 2 and cross(points[upper[-2]], points[upper[-1]], point) <= 0:
            upper.pop()
        upper.append(idx)
    
    # Remove last point of each half because it's repeated
    lower.pop()
    upper.pop()
    
    return lower + upper

def cross(o, a, b):
    """
    Cross product of vectors OA and OB.
    Positive if counter-clockwise, negative if clockwise, zero if collinear.
    """
    return (a[0] - o[0]) * (b[1] - o[1]) - (a[1] - o[1]) * (b[0] - o[0])
```
