def cross(O, A, B):
    return (A[0] - O[0]) * (B[1] - O[1]) - (A[1] - O[1]) * (B[0] - O[0])

def convex_hull(points):
    points = sorted(enumerate(points), key=lambda p: (p[1][0], p[1][1]))
    
    if len(points) <= 1:
        return points
    
    # Build lower hull
    lower = []
    for i, p in points:
        while len(lower) >= 2 and cross(lower[-2][1], lower[-1][1], p) <= 0:
            lower.pop()
        lower.append((i, p))
    
    # Build upper hull
    upper = []
    for i, p in reversed(points):
        while len(upper) >= 2 and cross(upper[-2][1], upper[-1][1], p) <= 0:
            upper.pop()
        upper.append((i, p))
    
    # Remove the last point of each half because it's repeated
    return lower[:-1] + upper[:-1]
