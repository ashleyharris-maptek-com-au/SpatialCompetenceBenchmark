import numpy as np
import math

def generate_pentagram_points(radius=2):
    """Generate vertices of a pentagram"""
    # Outer points
    outer = []
    for i in range(5):
        angle = math.radians(90 + i * 72)
        x = radius * math.cos(angle)
        y = radius * math.sin(angle)
        outer.append([x, y])
    
    # Inner points: intersections of star edges
    # Star edges: 0->2, 2->4, 4->1, 1->3, 3->0
    inner = []
    
    def line_intersection(p1, p2, p3, p4):
        x1, y1 = p1
        x2, y2 = p2
        x3, y3 = p3
        x4, y4 = p4
        
        denom = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4)
        if abs(denom) < 1e-10:
            return None
        
        t = ((x1 - x3) * (y3 - y4) - (y1 - y3) * (x3 - x4)) / denom
        
        x = x1 + t * (x2 - x1)
        y = y1 + t * (y2 - y1)
        return [x, y]
    
    # Find inner points by computing intersections of the star's edges
    star_connections = [[0, 2], [2, 4], [4, 1], [1, 3], [3, 0]]
    
    for i in range(5):
        # Get the current edge and the next edge
        edge_i = star_connections[i]
        p1, p2 = outer[edge_i[0]], outer[edge_i[1]]

        edge_next = star_connections[(i + 1) % 5]
        p3, p4 = outer[edge_next[0]], outer[edge_next[1]]

        intersection = line_intersection(p1, p2, p3, p4)
