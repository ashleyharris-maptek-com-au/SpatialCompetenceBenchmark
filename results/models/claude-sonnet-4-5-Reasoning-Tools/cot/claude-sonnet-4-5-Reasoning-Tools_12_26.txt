    dx_dy = [(1, 0), (0, 1), (-1, 0), (0, -1)]
    
    side_length = [12, 12, 12, 11]  # Number of 1m segments for each side of first rectangle
    
    pipes_placed = 0
    level = 0
    
    while pipes_placed < num_pipes:
        # Draw current rectangle
        for side in range(4):
            if pipes_placed >= num_pipes:
                break
            dx, dy = dx_dy[(direction + side) % 4]
            segments = side_length[side] - level
            for _ in range(segments):
                if pipes_placed >= num_pipes:
                    break
                x += dx
                y += dy
                points.append((x, y))
                pipes_placed += 1
        
        level += 1
        direction = (direction + 1) % 4
