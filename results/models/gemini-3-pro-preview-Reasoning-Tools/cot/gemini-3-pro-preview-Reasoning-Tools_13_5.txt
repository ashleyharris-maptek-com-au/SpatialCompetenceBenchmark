import numpy as np
import json

def is_blocked_aabb(target_point, sniper_pos, building_bounds):
    sx, sy, sz = sniper_pos
    tx, ty, tz = target_point
    b_xmin, b_xmax, b_ymin, b_ymax, b_zmin, b_zmax = building_bounds
    
    dx, dy, dz = tx - sx, ty - sy, tz - sz
    
    t_min = 0.0
    t_max = 1.0
    
    # Check X slabs
    if abs(dx) < 1e-9:
        if sx < b_xmin or sx > b_xmax: return False
    else:
        t1 = (b_xmin - sx) / dx
        t2 = (b_xmax - sx) / dx
        t_min = max(t_min, min(t1, t2))
        t_max = min(t_max, max(t1, t2))
        
    # Check Y slabs
    if abs(dy) < 1e-9:
        if sy < b_ymin or sy > b_ymax: return False
    else:
        t1 = (b_ymin - sy) / dy
        t2 = (b_ymax - sy) / dy
        t_min = max(t_min, min(t1, t2))
        t_max = min(t_max, max(t1, t2))
        
    # Check Z slabs
    if abs(dz) < 1e-9:
        if sz < b_zmin or sz > b_zmax: return False
    else:
        t1 = (b_zmin - sz) / dz
        t2 = (b_zmax - sz) / dz
        t_min = max(t_min, min(t1, t2))
        t_max = min(t_max, max(t1, t2))
        
    return t_max >= t_min

def is_blocked_aabb(target_point, sniper_pos, building_bounds):
    sx, sy, sz = sniper_pos
    tx, ty, tz = target_point
    b_xmin, b_xmax, b_ymin, b_ymax, b_zmin, b_zmax = building_bounds
    dx, dy, dz = tx - sx, ty - sy, tz - sz
    t_min, t_max = 0.0, 1.0
    
    for (o, d, b_min, b_max) in zip([sx, sy, sz], [dx, dy, dz], [b_xmin, b_ymin, b_zmin], [b_xmax, b_ymax, b_zmax]):
        if abs(d) < 1e-9:
            if o < b_min or o > b_max: return False
        else:
            t1, t2 = (b_min - o) / d, (b_max - o) / d
            t_min = max(t_min, min(t1, t2))
            t_max = min(t_max, max(t1, t2))
