// Pharaoh's Pyramid Stone Core Structure
// Optimized stepped pyramid model for marble facade planning
// Units: mm (SI metric, precise to 1mm)
// Each stone: 500mm cube
// Total stones used: 19019 (sum k^2 for k=1 to 38)
// Remaining stones: 981 (not used for height maximization)
// Layers: 38 (bottom: 38x38 stones, top: 1x1)
// Base side: 19000mm
// Total height: 19000mm
// Centered at origin (0,0,0), tomb below
// North: +y direction
// Base in XY plane, height +Z
// Smooth marble facade to cover steps for final pyramid shape

pyramid();

module pyramid() {
    for(i = [0 : 37]) {
        stones_side = 38 - i;
        side_mm = stones_side * 500;
        layer_bottom_z = i * 500;
        layer_center_z = layer_bottom_z + 250;
        translate([0, 0, layer_center_z])
            cube([side_mm, side_mm, 500], center = true);
    }
}
