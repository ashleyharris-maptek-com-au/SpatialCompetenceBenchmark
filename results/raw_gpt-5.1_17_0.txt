// Sand pile in silo - 1 unit = 1 meter

r_silo = 10*0.9144/2;
theta = 33;
V0 = 5*(0.9144^3);
V1 = 1500*(0.3048^3);
V2 = 100;
V3 = 20*44*0.00454609;
V = V0+V1+V2+V3;
r_natural = (3*V/tan(theta*PI/180)/PI)^(1/3);

module cone(r,h){cylinder(r1=r,r2=0,h=h,$fn=180);}

if (r_natural <= r_silo) {
  cone(r_natural, r_natural*tan(theta*PI/180));
} else {
  h_cyl = (V - PI*r_silo^3/3*tan(theta*PI/180))/(PI*r_silo^2);
  union() {
    translate([0,0,h_cyl]) cone(r_silo, r_silo*tan(theta*PI/180));
    cylinder(r=r_silo,h=h_cyl,$fn=180);
  }
}
