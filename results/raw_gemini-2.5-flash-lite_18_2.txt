                //     y_center = center_y

                current_layer_side_stones = BASE_LAYER_SIDE_STONES - z_layer_idx;
                current_layer_width_mm = current_layer_side_stones * STONE_SIDE_MM;
                offset_xz = (BASE_WIDTH_MM - current_layer_width_mm) / 2.0;
                center_y = STONE_SIDE_MM / 2.0 + z_layer_idx * STONE_SIDE_MM * 0.5; // Center of the stone in this layer along y

                // Iterate through stones in the current layer (i for x, j for z)
                for (i = 0; i < current_layer_side_stones; i++) {
                    for (j = 0; j < current_layer_side_stones; j++) {

                        // Calculate the center of the stone
                        stone_center_x = -BASE_WIDTH_MM / 2.0 + offset_xz + i * STONE_SIDE_MM + STONE_SIDE_MM / 2.0;
                        stone_center_z = -BASE_WIDTH_MM / 2.0 + offset_xz + j * STONE_SIDE_MM + STONE_SIDE_MM / 2.0;
                        stone_center_y = center_y;

                        // Place the stone. The `stone` module draws a cube at its translated origin.
                        // We want the center of the cube to be at (stone_center_x, stone_center_y, stone_center_z).
                        // So, we translate by [center_x - size/2, center_y - size/2, center_z - size/2].
                        translate([stone_center_x - STONE_SIDE_MM / 2.0,
                                   stone_center_y - STONE_SIDE_MM / 2.0,
                                   stone_center_z - STONE_SIDE_MM / 2.0]) {
                            cube([STONE_SIDE_MM, STONE_SIDE_MM, STONE_SIDE_MM]);
                        }
                    }
                }
            }
        }
    }
